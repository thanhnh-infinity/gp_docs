def qna_writer(description: str, status: str = "running") -> None:
    """
    Unified helper function to emit progress updates during Q&A workflow execution.

    Emits progress events using LangGraph's stream writer. Silently continues if
    stream writer is not available (e.g., running outside LangGraph context).

    Args:
        description: User-friendly description of the current step
        status: Status of the step ("running" or "completed"). Defaults to "running"

    Examples:
        >>> qna_writer("Semantic Query Analysis is running...")
        >>> qna_writer("We have the answer!", status="completed")
    """
    try:
        writer = get_stream_writer()
        progress_data = {
            "type": "progress",
            "description": description,
            "status": status,
        }
        writer(progress_data)
    except Exception as e:
        # Silently continue if stream writer not available (outside LangGraph context)
        logger.warning(f"qna_writer failed to emit progress: {e}")


def index_structured_outputs(
    outputs: List[PipelineStructuredOutput],
) -> Dict[str, PipelineStructuredOutput]:
    index: Dict[str, PipelineStructuredOutput] = {}

    def _walk(items: Iterable[PipelineStructuredOutput]) -> None:
        for o in items:
            if not isinstance(o, PipelineStructuredOutput):
                continue
            key = o.name
            if isinstance(key, str) and key and key not in index:
                index[key] = o
            subs = o.sub_section or []
            if subs:
                _walk(subs)

    _walk(outputs)
    return index


class BaseQAPipeline(ABC):
    """
    Abstract base class for workflow Q&A pipelines.

    The pipeline follows these steps:
    1. Semantic Query Planning - Determine what data to retrieve
    2. Structured Data Retrieval - Fetch relevant data based on plan
    3. Grounded Answer Generation - Generate answer from evidence
    """

    @abstractmethod
    def get_workflow_qa_specific_config(self) -> QAToolConfig:
        """Return the workflow-specific Q&A configuration."""
        pass

    @abstractmethod
    def get_workflow_id(self) -> str:
        pass

    async def solve(
        self,
        question: str,
        configuration: RunnableConfig,
    ) -> Tuple[str, List[PipelineStructuredOutput]]:
        start_time = perf_counter()
        generated_at = datetime.now()

        config = self.get_workflow_qa_specific_config()
        workflow_id = self.get_workflow_id()

        # Get run artifacts from store
        run_store = get_redis_run_store()
        thread_id = configuration["configurable"]["thread_id"]
        user_id = configuration["configurable"].get("user_id")

        run = await run_store.get_latest_success(
            thread_id=thread_id,
            workflow_id=workflow_id,
            user_id=user_id,
        )

        logger.info(
            f"[Q&A Framework Pipeline] Q&A for Workflow: {workflow_id} | Question: {question} | "
            f"Thread: {thread_id} | User: {user_id}"
        )

        if not run:
            logger.info(
                f"[Q&A Framework Pipeline] No workflow run found for thread={thread_id}. "
                f"Falling back to Perplexity for public data."
            )
            return await self._fallback_to_public_search(question)

        artifacts = run["artifacts"]
        structured_output = [
            PipelineStructuredOutput.model_validate(o)
            for o in artifacts["structured_output"]
        ]
        structured_output_by_id = index_structured_outputs(structured_output)
        ontology = artifacts.get("ontology")

        # 1) Semantic Query Planning: Build a semantic query plan based on the question, the ontology, and the structured output
        semantic_query_plan, reasoning_protocol = await self._build_query_plan(
            question=question,
            ontology=ontology,
            structured_output_by_id=structured_output_by_id,
            config=config,
        )

        # 2) Structured Data Retrieval: Retrieve the relevant structured data based on the semantic query plan and the structured output indexed by ID. This will return a list of PipelineStructuredOutput objects that match
        evidence_outputs = self._retrieve_evidence(
            structured_output_by_id=structured_output_by_id,
            semantic_query_plan=semantic_query_plan,
            config=config,
        )

        # 3) Grounded Answer Generation: Use the retrieved structured data and the question to generate a grounded answer using the GroundedAnswerEngine.
        answer_text = await self._generate_answer(
            question=question,
            evidence_outputs=evidence_outputs,
            ontology=ontology,
            reasoning_protocol=reasoning_protocol,
            config=config,
            workflow_id=workflow_id,
        )

        # Step 4: Final answer ready
        qna_writer("We have the answer, let me show you!", status="completed")

        answer_with_metadata = self._append_response_metadata(
            answer_text, generated_at, start_time
        )

        # TODO: Not sure that we need to return both content and artifact here, since the structured output is mainly for debugging/traceability purpose.
        # If we only expect that the Q&A response markdown only, the content is good enough. But if we want to enable the FE can display some cool things like viz table/chart, then we might need to return the structured output as well.
        return answer_with_metadata, []

    async def _build_query_plan(
        self,
        question: str,
        ontology: Optional[Dict],
        structured_output_by_id: Dict[str, PipelineStructuredOutput],
        config: QAToolConfig,
    ) -> Tuple[SemanticQueryPlan, Optional[object]]:
        """Build semantic query plan with optional reasoning protocol."""

        # Step 1: Emit progress - Semantic Query Planning
        qna_writer(
            "Semantic Query Planner Agent is running to understand your question and finding the content to do the reasoning and answer the question!"
        )

        # Create Semantic Query Planner using Factory
        planner = create_semantic_query_planner(
            config=config.semantic_query_planner_config,
            enable_aggregate_mode=config.enable_aggregate_mode,
            important_partitions=config.important_partitions,
            enable_important_partitions_fast_path=config.enable_important_partitions_fast_path,
        )
        designer = ReasoningProtocolDesigner(
            enable_reasoning_protocol=config.enable_reasoning_protocol
        )

        build_plan = planner.get_plan_function(
            use_batch_mode=config.use_selection_batch_mode
        )

        plan_kwargs = dict(
            question=question,
            ontology=ontology,
            structured_output_by_id=structured_output_by_id,
            suggested_selected_partitions=config.suggested_partitions,
            use_suggestions_only=config.use_suggestions_only,
            important_partitions=config.important_partitions,
            enable_important_partitions_fast_path=config.enable_important_partitions_fast_path,
        )

        reasoning_protocol = None

        """
        Run semantic query plan + reasoning protocol in PARALLEL to find:
        1. Which partitions/sections in the workflow output to answer the question
        2. Inside the Data-Set of each selected partition/section, which fields/columns to use to answer the question
        3. How many records /rows to retrieve from each selected partition/section to answer the question
        4. Which filtering methodologies should apply to filter the large data to retrieve set of enough records/rows to answer the question
        """

        if config.enable_reasoning_protocol:
            semantic_query_plan, reasoning_protocol = await asyncio.gather(
                build_plan(**plan_kwargs),
                designer.generate_ontology_reasoning_protocol(
                    question=question, ontology=ontology
                ),
                return_exceptions=True,
            )
            if isinstance(reasoning_protocol, Exception):
                logger.warning(
                    f"[Q&A] Reasoning protocol generation failed: {reasoning_protocol}"
                )
                reasoning_protocol = None
        else:
            semantic_query_plan: SemanticQueryPlan = await build_plan(**plan_kwargs)

        # POST-PROCESS: Apply working memory optimization
        semantic_query_plan = self._optimize_query_plan(semantic_query_plan, config)

        return semantic_query_plan, reasoning_protocol

    def _retrieve_evidence(
        self,
        structured_output_by_id: Dict[str, PipelineStructuredOutput],
        semantic_query_plan: SemanticQueryPlan,
        config: QAToolConfig,
    ) -> List[PipelineStructuredOutput]:
        """Retrieve structured data based on semantic query plan."""

        # Step 2: Emit progress - Getting necessary content
        qna_writer(
            "We have understood your question, the context and rules to reasoning and answer question. Let me query the real data to generate facts!"
        )

        retriever = StructuredDataRetriever(
            fail_open_on_bad_plan=config.fail_open_on_bad_plan
        )

        structured_evidence = retriever.retrieve(
            structured_output_by_id=structured_output_by_id,
            semantic_query_plan=semantic_query_plan,
        )

        return structured_evidence

    async def _generate_answer(
        self,
        question: str,
        evidence_outputs: List[PipelineStructuredOutput],
        ontology: Optional[Dict],
        reasoning_protocol: Optional[object],
        config: QAToolConfig,
        workflow_id: str,
    ) -> str:
        """Generate grounded answer from evidence."""

        # Step 3: Emit progress - Reasoning and finding answer
        qna_writer(
            "We are reasoning and finding the answer for your question. It will take some time!"
        )

        if (
            config.enable_reasoning_protocol
            and reasoning_protocol
            and hasattr(reasoning_protocol, "protocol_markdown")
            and reasoning_protocol.protocol_markdown
        ):
            reasoning_pattern_design = (
                DEFAULT_Q_AND_A_REASONING_PATTERN_DESIGN.rstrip()
                + "\n\n"
                + reasoning_protocol.protocol_markdown.strip()
                + "\n\n"
                + "You can now reason using the structured data below: \n"
            )
        else:
            # Seem like adding static default reasoning protocol here not help much on the performance, so comment it out for now.
            reasoning_pattern_design = (
                DEFAULT_Q_AND_A_REASONING_PATTERN_DESIGN.rstrip()
                + "\n\n"
                # + DEFAULT_REASONING_PROTOCOL_PATTERN_DESIGN.strip()
                # + "\n\n"
                + "You can now reason using the structured data below: \n"
            )

        answer_engine = GroundedAnswerEngine(workflow_id=workflow_id)
        return await answer_engine.answer(
            reasoning_pattern_design=reasoning_pattern_design,
            question=question,
            evidence_outputs=evidence_outputs,
            ontology=ontology,
            include_ontology=config.include_ontology_in_answer,
        )

    @staticmethod
    def _optimize_query_plan(
        plan: SemanticQueryPlan,
        config: QAToolConfig,
    ) -> SemanticQueryPlan:
        """
        Post-process semantic query plan for working memory optimization.

        That could handle MANY MANY THINGS related to working memory optimization in the future.
        Optimizations applied:
        1. Shared data partitions: skip duplicate data when both primary/secondary selected
        2. Partition max_rows override: increase max_rows for partitions needing full scan
        """
        if not plan.sections:
            return plan

        # We need a map of partition_id -> SectionSelectionPlan to faster lookup
        section_map: Dict[str, SectionSelectionPlan] = {
            sp.id: sp for sp in plan.sections if sp.id
        }

        # Optimization 1: Shared Data Partitions
        if config.shared_data_partitions:
            active_partitions: set[str] = {
                sp.id
                for sp in plan.sections
                if sp.id and sp.record_plan and sp.record_plan.mode != "none"
            }

            for secondary_id, primary_id in config.shared_data_partitions.items():
                if (
                    secondary_id in active_partitions
                    and primary_id in active_partitions
                ):
                    section = section_map.get(secondary_id)
                    if section and section.record_plan:
                        original_mode = section.record_plan.mode
                        section.record_plan.mode = "none"
                        section.record_plan.skip_reason = (
                            f"shared_data_with:{primary_id}"
                        )
                        logger.info(
                            f"[Q&A Planner Optimizer] Working memory optimization applied: "
                            f"partition={secondary_id} mode changed {original_mode} -> none "
                            f"(shares data with {primary_id})"
                        )

        # Optimization 2: Partition Max Rows Override
        if config.partition_max_rows:
            for partition_id, max_rows in config.partition_max_rows.items():
                section = section_map.get(partition_id)
                if section and section.record_plan:
                    # Only apply for mode="all" - other modes have their own row control
                    if section.record_plan.mode == "all":
                        original_max_rows = section.record_plan.max_rows
                        section.record_plan.max_rows = max_rows
                        logger.info(
                            f"[Q&A Planner Optimizer] Working memory optimization applied for Max_rows: "
                            f"partition={partition_id} max_rows changed {original_max_rows} -> {max_rows}"
                        )

        return plan

    @staticmethod
    def _append_response_metadata(
        answer_text: str,
        generated_at: datetime,
        start_time: float,
    ) -> str:
        elapsed_seconds = perf_counter() - start_time
        timestamp_str = generated_at.strftime("%Y-%m-%d %H:%M:%S")
        if elapsed_seconds < 1:
            elapsed_str = f"{elapsed_seconds * 1000:.0f}ms"
        elif elapsed_seconds < 60:
            elapsed_str = f"{elapsed_seconds:.1f}s"
        else:
            minutes = int(elapsed_seconds // 60)
            seconds = elapsed_seconds % 60
            elapsed_str = f"{minutes}m {seconds:.1f}s"
        metadata_footer = (
            f"\n\n---\n"
            f"*Response generated at {timestamp_str} | Response time: {elapsed_str}*"
        )

        return answer_text.rstrip() + metadata_footer

    @staticmethod
    async def _fallback_to_public_search(
        question: str,
    ) -> Tuple[str, List[PipelineStructuredOutput]]:
        """
        Fallback to Perplexity when question is out of workflow scope.
        Returns answer from public data with clear disclaimer.
        """
        # Disclaimer header
        disclaimer = (
            "---\n"
            "I am part of the Growth Protocol (GP) Core AI Reasoning Engine, which is the reasoning component of the GP AI Platform. \n\n"
            "**Important Notice:** This response is generated from **publicly available data** only. "
            "No proprietary, confidential, or workflow-specific data has been used or accessed.\n\n"
            "---\n\n"
            "**For workflow workflow authorized user**: If you are in the workflow and cannot access your proprietary data, it seems that the session has ended. Please re-run the workflow and ask your question again. \n\n"
        )

        try:
            result = await ask_perplexity_async(question)
            answer = result.get("answer", "")
            citations = result.get("citations", [])
            if not answer:
                full_response = (
                    disclaimer
                    + "I couldn't find relevant public information to answer your question. "
                    "Please try rephrasing your question or ask about a topic that's available in public sources."
                )
                return full_response, []

            # Build response with sources
            response_parts = [disclaimer, answer]
            if citations:
                response_parts.append("\n\n---\n")
                response_parts.append("**Sources & References:**\n")
                for i, citation in enumerate(citations, 1):
                    if isinstance(citation, str):
                        response_parts.append(f"{i}. {citation}\n")
                    elif isinstance(citation, dict):
                        url = citation.get("url", citation.get("link", ""))
                        title = citation.get("title", url)
                        response_parts.append(f"{i}. [{title}]({url})\n")
            full_response = "".join(response_parts)
            logger.info(
                f"[Q&A Framework Pipeline] Perplexity fallback returned answer with "
                f"{len(citations)} citations"
            )
            return full_response, []

        except Exception as e:
            logger.error(f"[Q&A Framework Pipeline] Perplexity fallback failed: {e}")
            error_response = (
                disclaimer
                + "I was unable to retrieve public information at this time. "
                "Please try again later or contact support if the issue persists.\n\n"
                f"*Error details: {str(e)}*"
            )
            return error_response, []

class QuestionAnsweringFrameworkVersion(str, Enum):

class SemanticQueryPlannerStrategy(str, Enum):
    """Strategy for semantic query planning."""

    # Use LLM with Ontology-driven + rules
    LLM_ONTOLOGY_DRIVEN = "llm_ontology_driven"
    # Use deterministic search with Ontology-driven/indexing + rules (completely Symbolic)
    DETERMINISTIC = "deterministic"
    # Try deterministic first, fallback to LLM if low confidence
    HYBRID = "hybrid"


@dataclass
class SemanticQueryPlannerConfig:
    strategy: SemanticQueryPlannerStrategy = (
        SemanticQueryPlannerStrategy.LLM_ONTOLOGY_DRIVEN
    )
    # LLM_ONTOLOGY_DRIVEN strategy
    main_llm_model: Optional[str] = LLMVertexAIGeminiModels.GEMINI_2_5_FLASH.value
    fallback_llm_models: Optional[List[str]] = None
    temperature: float = 0.0
    max_output_tokens: int = 100_000
    max_input_tokens: int = 100_000
    agent_request_time_out: int = 60
    max_attempts: int = 2

    # Deterministic Configuration (for DETERMINISTIC strategy)
    # --- Section Selection Settings ---
    # Enable sentence-transformer embeddings (requires: pip install sentence-transformers)
    deterministic_enable_embeddings: bool = True
    # Embedding model name (e.g., "all-MiniLM-L6-v2", "all-mpnet-base-v2")
    deterministic_search_index: Optional[str] = None
    # Weight for lexical (BM25) score in combined ranking
    deterministic_lexical_weight: float = 0.6
    # Weight for embedding similarity score in combined ranking
    deterministic_embedding_weight: float = 0.4
    # Maximum sections to return from search
    deterministic_top_k_sections: int = 10
    # Minimum score threshold for section selection
    deterministic_min_section_score: float = 0.05
    # Confidence threshold for accepting deterministic results (used in HYBRID)
    deterministic_confidence_threshold: float = 0.8
    # --- Field Selection Settings ---
    # Maximum fields to select per section
    deterministic_max_fields_per_section: int = 30
    # Minimum score threshold for field selection
    deterministic_min_field_score: float = 0.1
    # Weight for token overlap scoring
    deterministic_field_token_weight: float = 0.3
    # Weight for synonym matching scoring
    deterministic_field_synonym_weight: float = 0.2
    # Weight for ontology description matching scoring
    deterministic_field_description_weight: float = 0.3
    # Weight for field importance heuristics (id, name, date fields)
    deterministic_field_importance_weight: float = 0.2
    # --- Record Plan Settings ---
    # Default K for top_k mode
    deterministic_default_k: int = 20
    # Row count threshold for using mode="all" automatically
    deterministic_small_section_threshold: int = 50
    # Row count threshold for medium sections
    deterministic_medium_section_threshold: int = 100
    # --- Custom Synonyms ---
    # Extend default synonyms with domain-specific mappings
    # Example: {"dc": {"distribution_center", "warehouse"}, "ttm": {"trailing_twelve_months"}}
    deterministic_custom_synonyms: Optional[Dict[str, Set[str]]] = None
    deterministic_rules_path: Optional[str] = None

    # Hybrid Configuration (for HYBRID strategy)
    # Reserved for future implementation
    hybrid_fallback_to_llm: bool = True
    hybrid_llm_threshold: float = 0.7


@dataclass
class QAToolConfig:
    # Sub-Strategy-Configuration for Semantic Query Planner
    semantic_query_planner_config: SemanticQueryPlannerConfig = field(
        default_factory=SemanticQueryPlannerConfig
    )

    # Reasoning protocol
    enable_reasoning_protocol: bool = (
        False  # True = dynamic reasoning-protocol-generated, False = static
    )

    # Section selection
    use_suggestions_only: bool = (
        True  # True = use predefined sections, False = smart sections selection
    )

    suggested_partitions: Optional[List[str]] = None  # Predefined section IDs

    # Query plan mode
    use_selection_batch_mode: bool = True  # True = batch (fewer fields and rows collecting processes), False = parallel (more granular)

    # Retriever behavior
    fail_open_on_bad_plan: bool = (
        False  # True = return all data on bad plan, False = strict
    )

    # Answer generation
    include_ontology_in_answer: bool = False  # Pass ontology context to final answer

    # Important partitions that should always use mode="all" (like summary or analytic tables)
    important_partitions: Optional[List[str]] = None

    # Fast path for important partitions: True = skip LLM for important partitions (all fields + mode="all")
    #                                     False = LLM selects fields, but still force mode="all" for important partitions
    enable_important_partitions_fast_path: bool = True

    # Aggregate mode control - disabled by default for backward compatibility and performance monitoring
    enable_aggregate_mode: bool = False

    # ============================================================
    # Shared Data Partitions (Working Memory Optimization)
    # ============================================================
    # Maps secondary partition -> primary partition when they share the same dataset.
    # When BOTH partitions are selected in the query plan, the secondary partition's
    # data will be skipped (uses primary's data only) to avoid duplication.
    # When only ONE is selected, normal behavior applies.
    #
    # Format: {secondary_partition_id: primary_partition_id}
    # Example: {"competitive_pricing_geographic": "competitive_pricing_between_company_and_competitors"}
    shared_data_partitions: Optional[Dict[str, str]] = None

    # Working Memory Optimization - Partition Max Rows Override
    # Override default max_rows (200) for specific partitions.
    # Use when certain partitions need full data access for accurate answers.
    # Format: {partition_id: max_rows}
    # Example: {"price_recommendations": 1000}
    # WARNING: Higher values increase token usage and latency.
    # Only use for partitions where questions require full dataset analysis
    # (e.g., "which has highest/lowest", "across all markets").
    partition_max_rows: Optional[Dict[str, int]] = None

class OperationSet:
    def __init__(self, *values: str):
        self._values: Tuple[str, ...] = values

    @property
    def values(self) -> Tuple[str, ...]:
        return self._values

    def to_list(self) -> list:
        return list(self._values)

    def to_bnf_choice(self) -> str:
        """Return pipe-separated string  E.g., all|top_k|filter|none"""
        # Represent the content as <exp> ::= A | B | C
        return "|".join(self._values)

    def to_quoted_bnf_choice(self) -> str:
        """Return pipe-separated string with quotes. E.g., "all"|"top_k"|"filter"|"none" """
        return "|".join(f'"{v}"' for v in self._values)

    def __contains__(self, item: str) -> bool:
        return item in self._values

    def __iter__(self):
        return iter(self._values)

    def __len__(self) -> int:
        return len(self._values)

      - Price variance and price architecture questions
      - Competitive positioning and pricing strategy questions
      - Elasticity and price sensitivity analysis
      - Price recommendation queries
      - Gross profit pool and margin share analysis
      - Sentiment analysis from social platforms (Twitter, Reddit) associated with Pricing Competitive Intelligence Workflow
      - Cannibalization and margin leakage identification
      - Rebate percentage analysis by market

    **IMPORTANT - ALSO USE THIS TOOL FOR:**
      - "How to improve" questions that need data context (e.g., "How could we improve the price recommendation engine?")
      - Strategy recommendations that should be grounded in actual data
      - Questions asking for actionable insights based on the pricing data
      - "What should we do" or "What would you recommend" questions
      - Any question where answering requires examining the workflow's pricing data first

    **WHEN NOT TO USE THIS TOOL (LLM should answer directly):**
      - Meta questions about methodology or assumptions
      - General explanations about pricing concepts
      - Clarifications about the analysis approach

    Example questions that SHOULD trigger this tool:
      - "Which SKUs have the widest price range?"
      - "What is the average elasticity of the market?"
      - "Which market has the highest rebate %?"
      - "Where is cannibalization eating gross profit?"
      - "What actionable insights can I get to dominate A.O. Smith"
      - "How could we improve the price recommendation engine?"
      - "Based on the data, what pricing strategy should we use?"

    Example questions that SHOULD NOT trigger this tool:
      - "Explain how elasticity is calculated"
      - "What assumptions are in the price recommendation engine?"
    """
    # Make sure to get the original question from metadata if exists LLM hallucinate a LOT
    original_question = (
        configuration.get("metadata", {}).get("original_question") or question
    )
    pipeline = PricingCompetitiveIntelligenceQAPipeline()
    return await pipeline.solve(question=original_question, configuration=configuration)

class DeterministicQueryPlanner:
    """
    Deterministic query planner using search and rules.

    Components:
    1. Search Index: Lexical + optional embedding search for section selection
    2. Field Selector: Token/synonym/description matching for field selection
    3. Record Planner: Rule-based intent detection for record plan
    """

    def __init__(
        self,
        config: Optional[DeterministicQueryPlannerConfig] = None,
    ):
        self.config = config or DeterministicQueryPlannerConfig()

        # Initialize search index
        self.search_index = DeterministicSearchIndex(
            enable_embeddings=self.config.enable_embeddings,
            embedding_model=self.config.embedding_model,
            lexical_weight=self.config.lexical_weight,
            embedding_weight=self.config.embedding_weight,
        )

        # Initialize field selector with synonyms
        synonyms = DEFAULT_SYNONYMS.copy()
        if self.config.custom_synonyms:
            synonyms.update(self.config.custom_synonyms)

        self.field_selector = DeterministicFieldSelector(
            synonyms=synonyms,
            token_weight=self.config.field_token_weight,
            synonym_weight=self.config.field_synonym_weight,
            description_weight=self.config.field_description_weight,
            importance_weight=self.config.field_importance_weight,
            min_score_threshold=self.config.min_field_score,
            max_fields=self.config.max_fields_per_section,
        )

        # Initialize record planner
        self.record_planner = DeterministicRecordPlanner(
            enable_aggregate_mode=self.config.enable_aggregate_mode,
            default_k=self.config.default_k,
            small_section_threshold=self.config.small_section_threshold,
            medium_section_threshold=self.config.medium_section_threshold,
        )

        self._index_built = False
        self._last_ontology_hash: Optional[int] = None

    def _get_ontology_hash(self, ontology: Dict[str, Any]) -> int:
        """Get a simple hash to detect ontology changes."""
        return hash(frozenset(ontology.keys()))

    def _ensure_index(self, ontology: Dict[str, Any]) -> None:
        """Build index if not already built or ontology changed."""
        current_hash = self._get_ontology_hash(ontology)
        if not self._index_built or self._last_ontology_hash != current_hash:
            self.search_index.build_index(ontology)
            self._index_built = True
            self._last_ontology_hash = current_hash

    def _select_sections(
        self,
        question: str,
        ontology: Dict[str, Any],
        suggested_partitions: Optional[List[str]] = None,
        use_suggestions_only: bool = False,
    ) -> List[str]:
        """Select relevant sections using search."""

        # If using suggestions only, return them directly
        if use_suggestions_only and suggested_partitions:
            allowed_set = set(ontology.keys())
            return [sid for sid in suggested_partitions if sid in allowed_set]

        # Build/rebuild index if needed
        self._ensure_index(ontology)

        # Search for relevant sections
        results = self.search_index.search(
            query=question,
            top_k=self.config.top_k_sections,
            min_score=self.config.min_section_score,
        )

        selected_sids = [r.sid for r in results]

        # Merge with suggested partitions (suggested first)
        if suggested_partitions:
            allowed_set = set(ontology.keys())
            suggested_valid = [
                sid for sid in suggested_partitions if sid in allowed_set
            ]
            merged = suggested_valid + [
                sid for sid in selected_sids if sid not in set(suggested_valid)
            ]
            selected_sids = merged

        logger.info(
            f"[DeterministicQueryPlanner] Selected sections: {selected_sids[:10]}"
            f"{' (truncated)' if len(selected_sids) > 10 else ''}"
        )

        return selected_sids

    def _build_section_plan(
        self,
        question: str,
        section_id: str,
        structured_output: Any,
        ontology: Dict[str, Any],
    ) -> Optional[SectionSelectionPlan]:
        """Build plan for a single section."""

        # Extract data from structured output
        data = (
            getattr(structured_output, "data", None)
            if not isinstance(structured_output, dict)
            else structured_output.get("data")
        )

        if not isinstance(data, list) or not data:
            return None

        # Extract headers from first row
        headers = data[0] if isinstance(data[0], list) else []
        allowed_fields = [h for h in headers if isinstance(h, str)]

        if not allowed_fields:
            return None

        row_count = max(0, len(data) - 1)

        # Get section ontology
        section_ontology = ontology.get(section_id, {})
        if not isinstance(section_ontology, dict):
            section_ontology = {}

        # Fast path for important partitions
        if (
            self.config.enable_important_partitions_fast_path
            and self.config.important_partitions
            and section_id in self.config.important_partitions
        ):
            fields = allowed_fields[: self.config.max_fields_per_section]
            record_plan = RecordPlan(mode="all")
            logger.info(
                f"[DeterministicQueryPlanner] Important partition '{section_id}': "
                f"all {len(fields)} fields, mode=all"
            )
            return SectionSelectionPlan(
                id=section_id,
                fields=fields,
                record_plan=record_plan,
            )

        # Select fields using field selector
        selected_fields, _ = self.field_selector.select_fields(
            question=question,
            allowed_fields=allowed_fields,
            section_ontology=section_ontology,
        )

        if not selected_fields:
            return SectionSelectionPlan(
                id=section_id,
                fields=[],
                record_plan=RecordPlan(mode="none"),
            )

        # Plan records using record planner
        record_plan_dict = self.record_planner.plan_records(
            question=question,
            section_id=section_id,
            selected_fields=selected_fields,
            allowed_fields=allowed_fields,
            row_count=row_count,
        )

        # Convert dict to RecordPlan model
        record_plan = self._dict_to_record_plan(record_plan_dict, allowed_fields)

        logger.debug(
            f"[DeterministicQueryPlanner] section={section_id}, "
            f"fields={len(selected_fields)}, mode={record_plan.mode}"
        )

        return SectionSelectionPlan(
            id=section_id,
            fields=selected_fields,
            record_plan=record_plan,
        )

    def _dict_to_record_plan(
        self,
        plan_dict: Dict[str, Any],
        allowed_fields: List[str],
    ) -> RecordPlan:
        """Convert dict to RecordPlan model."""
        mode = plan_dict.get("mode", "none")
        allowed_set = set(allowed_fields)

        plan = RecordPlan(mode=mode)

        if mode == "top_k":
            plan.k = plan_dict.get("k", 20)
            sort_by = plan_dict.get("sort_by")
            if sort_by and sort_by in allowed_set:
                plan.sort_by = sort_by
            plan.descending = plan_dict.get("descending", True)

        elif mode == "all":
            max_rows = plan_dict.get("max_rows")
            if isinstance(max_rows, int) and 1 <= max_rows <= 1000:
                plan.max_rows = max_rows

        elif mode == "aggregate":
            aggregations = plan_dict.get("aggregations", [])
            if aggregations:
                valid_aggs = []
                for agg in aggregations:
                    agg_field = agg.get("field", "")
                    if agg_field in allowed_set:
                        valid_aggs.append(
                            Aggregation(
                                field=agg_field,
                                op=agg.get("op", "count"),
                                alias=agg.get("alias"),
                            )
                        )
                plan.aggregations = valid_aggs if valid_aggs else None

            group_by = plan_dict.get("group_by", [])
            if group_by:
                valid_group_by = [g for g in group_by if g in allowed_set]
                plan.group_by = valid_group_by if valid_group_by else None

        elif mode == "filter":
            filters = plan_dict.get("filters", [])
            if filters:
                valid_filters = []
                for f in filters:
                    f_field = f.get("field", "")
                    if f_field in allowed_set:
                        valid_filters.append(
                            RecordFilter(
                                field=f_field,
                                op=f.get("op", "=="),
                                value=f.get("value"),
                            )
                        )
                plan.filters = valid_filters if valid_filters else None

        return plan

    async def build_semantic_query_plan(
        self,
        *,
        question: str,
        ontology: Dict[str, Any],
        structured_output_by_id: Dict[str, Any],
        suggested_selected_partitions: Optional[List[str]] = None,
        use_suggestions_only: bool = False,
        important_partitions: Optional[List[str]] = None,
        enable_important_partitions_fast_path: bool = True,
    ) -> SemanticQueryPlan:
        """
        Build semantic query plan using deterministic methods.

        This is the main entry point, compatible with LLM planner interface.
        """
        # Update config with runtime parameters
        if important_partitions:
            self.config.important_partitions = important_partitions
        self.config.enable_important_partitions_fast_path = (
            enable_important_partitions_fast_path
        )

        # Step 1: Select sections
        section_ids = self._select_sections(
            question=question,
            ontology=ontology,
            suggested_partitions=suggested_selected_partitions,
            use_suggestions_only=use_suggestions_only,
        )

        # Step 2: Build plans for each section
        plans: List[SectionSelectionPlan] = []

        for sid in section_ids:
            so = structured_output_by_id.get(sid)
            if not so:
                continue

            plan = self._build_section_plan(
                question=question,
                section_id=sid,
                structured_output=so,
                ontology=ontology,
            )

            if plan:
                logger.info(
                    f"[DeterministicQueryPlanner - Leve2+3 - BatchMode] section={sid} ; fields={plan.fields} ; record_plan={plan.record_plan}"
                )
                plans.append(plan)

        logger.info(
            f"[DeterministicQueryPlanner] Built plan with {len(plans)} sections"
        )

        return SemanticQueryPlan(sections=plans)

    async def build_semantic_query_plan_batch(
        self,
        *,
        question: str,
        ontology: Dict[str, Any],
        structured_output_by_id: Dict[str, Any],
        suggested_selected_partitions: Optional[List[str]] = None,
        use_suggestions_only: bool = False,
        important_partitions: Optional[List[str]] = None,
        enable_important_partitions_fast_path: bool = True,
    ) -> SemanticQueryPlan:
        """
        Batch mode - same as regular mode for deterministic planner.

        For deterministic planner, batch and regular mode are identical
        since there's no LLM call batching benefit.
        """
        return await self.build_semantic_query_plan(
            question=question,
            ontology=ontology,
            structured_output_by_id=structured_output_by_id,
            suggested_selected_partitions=suggested_selected_partitions,
            use_suggestions_only=use_suggestions_only,
            important_partitions=important_partitions,
            enable_important_partitions_fast_path=enable_important_partitions_fast_path,
        )

class GroundedAnswerEngine:
    def __init__(
        self,
        cfg: Optional[GroundedAnswerEngineConfig] = None,
        workflow_id: Optional[str] = None,
    ):
        self.cfg = cfg or GroundedAnswerEngineConfig()
        self.workflow_id = workflow_id

    @staticmethod
    def _build_system_prompt_grounded_reasoning_answer() -> str:
        return """\
You are a grounded reasoning and answer engine for a workflow Q&A tool. 
Your task is to provide accurate, concise, and business-focused answers to user questions based Reasoning over the structured evidence data and ontology-driven.
Your audiences are senior leadership, managers, executives, business stakeholders and decision makers.

Non-negotiable rules:
1) Use ONLY the provided structured evidence in the prompt. No external knowledge.
2) NEVER fabricate fields/rows/values. If evidence is insufficient, say so.
3) NEVER contradict explicit values in the evidence.
4) MUST use the ontology (if present) as ground truth for interpreting data fields.
5) Do NOT output your internal chain-of-thought. Provide only a concise reasoning summary.

Communication style (CRITICAL):
- Write for executives, managers, and business stakeholders - NOT technical audiences.
- Lead with the business impact and key takeaways first.
- Use clear, business-friendly language. Avoid jargon, technical terms, and abbreviations unless industry-standard.
- Translate data insights into actionable business implications (e.g., "revenue opportunity", "cost savings", "risk exposure").
- When presenting numbers, provide business context (e.g., "This represents a 15% improvement over baseline" rather than just "Value: 1.15").
- Use currency formatting for dollar amounts (e.g., "$1.2M" not "1200000").
- Round percentages appropriately (e.g., "12.5%" not "12.4837%").

Output:
- Return a single well-structured Markdown answer.
- Start with a brief executive summary (1-2 sentences) when answering complex questions.
- Prefer bullet points / numbered steps / small tables where helpful.
- Highlight key business metrics and their implications.
- If you make assumptions, label them explicitly as `Assumptions`.
"""

    async def answer(
        self,
        *,
        reasoning_pattern_design: str,
        question: str,
        evidence_outputs: List[PipelineStructuredOutput],
        ontology: Optional[Dict[str, Any]] = None,
        include_ontology: bool = False,
    ) -> str:
        if not question or not question.strip():
            return "No question provided."

        if not evidence_outputs:
            return (
                "I don't have any workflow evidence to answer this question. "
                "Please run the workflow first (or adjust retrieval)."
            )

        # Simple routing: complex questions --> then High Thinking Model, simple --- Fast Model
        router = RouterAgent(workflow_id=self.workflow_id)
        use_reasoning_model = router._requires_complex_reasoning(question)
        selected_model = (
            self.cfg.reasoning_llm_model
            if use_reasoning_model
            else self.cfg.main_llm_model
        )
        logger.info(
            f"[Q&A Framework GroundedAnswerEngine] Question routing: "
            f"workflow={self.workflow_id}, complex={use_reasoning_model}, "
            f"model={selected_model.split('/')[-1]}"
        )

        system_prompt = self._build_system_prompt_grounded_reasoning_answer()
        user_prompt = build_grounded_reasoning_user_prompt(
            reasoning_pattern_design=reasoning_pattern_design,
            question=question,
            evidence_outputs=evidence_outputs,
            ontology=ontology,
            include_ontology=include_ontology,
        )

        token_size = count_tokens(user_prompt) + count_tokens(system_prompt)
        logger.info(
            f"[Q&A Framework GroundedAnswerEngine] Total prompt tokens: {token_size}"
        )
        if token_size > self.cfg.max_input_tokens:
            logger.warning(
                f"[Q&A Framework GroundedAnswerEngine] Prompt token size {token_size} exceeds max_input_tokens {self.cfg.max_input_tokens}"
            )
            return "Question and Answering context is not available at the moment. Can you please try again with a different question?"

        last_raw = ""
        for attempt in range(self.cfg.max_attempts):
            raw = await llmus.get_llm_response_with_backoff(
                model=selected_model,
                system_prompt=system_prompt.strip(),
                user_prompt=user_prompt.strip()
                if attempt == 0
                else self._repair_user_prompt(user_prompt.strip(), last_raw),
                temperature=self.cfg.temperature,
                caching=True,
                request_timeout=self.cfg.agent_request_time_out,
                fallback_models=self.cfg.fallback_llm_models,
            )
            last_raw = (raw or "").strip()
            if last_raw:
                return last_raw

        return "I couldn't produce an answer from the provided evidence."

    @staticmethod
    def _repair_user_prompt(original_user_prompt: str, last_raw: str) -> str:
        repair_prefix = (
            "Your previous answer was empty or invalid. "
            "Return a concise grounded answer using ONLY the evidence below.\n\n"
        )
        prev = (last_raw or "")[:2000]
        return f"{repair_prefix}Previous output:\n{prev}\n\nEVIDENCE:\n{original_user_prompt}"

class ReasoningProtocolDesigner:
    fallback = """\
### Reasoning Protocol

Follow these steps internally:
1) Identify which evidence_outputs sections contain relevant fields/rows for the question.
2) Extract only explicit values from evidence_outputs; do not infer missing numbers.
3) If the evidence is insufficient, say what is missing.

Output:
- Provide a concise Markdown answer.
- Use bullets and small tables when helpful.
- Label any assumptions as `Assumptions`.
- Do NOT reveal chain-of-thought; provide only a brief reasoning summary.
"""

    def __init__(
        self,
        config: Optional[ReasoningProtocolDesignerConfig] = None,
        enable_reasoning_protocol: Optional[bool] = None,
    ):
        self.cfg = config or ReasoningProtocolDesignerConfig()
        if enable_reasoning_protocol is not None:
            self.cfg.enable_reasoning_protocol = enable_reasoning_protocol
        logger.info(
            f"[Q&A Reasoning Protocol Designer] Generating ontology-aware reasoning protocol ENABLED={self.cfg.enable_reasoning_protocol}"
        )

    @staticmethod
    def _build_system_prompt_reasoning_protocol() -> str:
        return """\
You are a prompt synthesizer or a reasoning protocol generator for a grounded workflow Q&A system.

Goal:
- Produce a compact "Reasoning Protocol" prompt that will be appended to a default grounding prefix.
- The runtime reasoner will receive:
  (a) the user question
  (b) structured evidence outputs
  (c) ontology_context (already provided elsewhere)
- Your protocol must guide the runtime reasoner to answer accurately using ONLY the provided evidence and ontology.

Non-negotiable constraints:
1) Output MUST be valid JSON only (no markdown outside JSON).
2) Focus on HOW to extract and compare data, not WHAT the data means
3) Output schema MUST be exactly:
   {"protocol_markdown": "<MARKDOWN STRING>"}
4) The value of "protocol_markdown" MUST be valid Markdown (headings, bullets, steps, tables allowed).
5) Do NOT paste the full ontology. Refer to "ontology_context" as already present.
6) Do NOT ask the runtime model to reveal chain-of-thought. Instruct it to reason internally and output only a concise reasoning summary.
7) The protocol_markdown must be short (max 500 words), actionable, and optimized for reliability (guardrails, steps, formats, and failure handling).
"""

    @staticmethod
    def _build_user_prompt_reasoning_protocol(
        *,
        question: str,
        ontology_compact: Dict[str, Any],
        token_budget: int,
    ) -> str:
        """
        We pass a compact ontology view to let the synthesizer understand:
        - what sections exist
        - what metadata/fields mean (at high-level)
        Without duplicating full ontology in the final protocol.
        """
        return json.dumps(
            {
                "question": question,
                "constraints": [
                    "Protocol should assume ontology_context will be provided at runtime; do not duplicate it.",
                    "Protocol must enforce: use ONLY evidence_outputs + ontology_context; never external knowledge.",
                    "Protocol must enforce: never fabricate values; if insufficient evidence, say so.",
                    "Protocol should recommend internal reasoning techniques (CoT/ToT/self-consistency) but never output chain-of-thought.",
                    "Keep protocol concise to reduce token usage.",
                ],
                "available_ontology_compact": ontology_compact,
                "preferred_techniques_pool": [
                    "cot_internal",
                    "tot_internal",
                    "self_consistency_light",
                    "few_shot_mini",
                    "zero_shot_strict",
                ],
                "token_budget_for_protocol_markdown": token_budget,
                "output_schema": {
                    "protocol_markdown": "<markdown string to append>",
                    # "techniques": ["<string>", "..."],
                    # "notes": {"token_budget": token_budget, "style": "concise"},
                },
            },
            ensure_ascii=False,
        )

    @staticmethod
    def _validate_reasoning_protocol_payload(
        payload: dict,
    ) -> Optional[GroundedReasoningProtocolArtifact]:
        if not isinstance(payload, dict):
            return None

        protocol = payload.get("protocol_markdown")
        if not isinstance(protocol, str) or not protocol.strip():
            return None

        techniques = payload.get("techniques")
        if techniques is not None:
            if not isinstance(techniques, list) or any(
                not isinstance(x, str) for x in techniques
            ):
                techniques = None

        notes = payload.get("notes")
        if notes is not None and not isinstance(notes, dict):
            notes = None

        # Hard trim to keep it compact, just in case
        protocol = protocol.strip()
        if len(protocol) > 12000:
            protocol = protocol[:12000].rstrip() + "\n..."

        return GroundedReasoningProtocolArtifact(
            protocol_markdown=protocol,
            techniques=techniques,
            metadata=notes,
        )

    @staticmethod
    def _build_system_prompt_protocol_judge() -> str:
        return """\
You are a strict judge. Pick the best protocol among candidates for reliability and token efficiency.

Rules:
1) Output JSON only: {"best_index": <int>, "why": "<short>"}
2) Prefer protocols that:
   - enforce grounded-only evidence
   - instruct internal reasoning but forbid exposing chain-of-thought
   - specify a clear answer format
   - handle insufficient evidence safely
   - are concise
"""

    @staticmethod
    def _build_user_prompt_protocol_judge(
        *,
        question: str,
        candidates: List[str],
    ) -> str:
        return json.dumps(
            {
                "question": question,
                "candidates": [
                    {"index": i, "protocol_markdown": c}
                    for i, c in enumerate(candidates)
                ],
                "scoring_criteria": [
                    "grounded_only_enforcement",
                    "no_chain_of_thought_exposure",
                    "clear_steps_and_format",
                    "safe_failure_mode",
                    "concise",
                ],
                "output_schema": {"best_index": 0, "why": "short"},
            },
            ensure_ascii=False,
        )

    async def generate_ontology_reasoning_protocol(
        self,
        *,
        question: str,
        ontology: Dict[str, Any],
    ) -> GroundedReasoningProtocolArtifact:
        """
        Generates a compact, ontology-aware reasoning protocol that will be appended to
        DEFAULT_Q_AND_A_REASONING_PATTERN_DESIGN.

        Key property:
        - DOES NOT include the full ontology. It assumes ontology_context will be provided
          by GroundedAnswerEngine via include_ontology=True.
        """

        logger.info(
            f"Generating ontology-aware reasoning protocol for question: {question}"
        )

        if not question or not question.strip():
            return GroundedReasoningProtocolArtifact(
                protocol_markdown="",
                techniques=["zero_shot_strict"],
                metadata={"error": "empty_question"},
            )

        if not isinstance(ontology, dict) or not ontology:
            # If no ontology, still produce a grounded protocol (evidence only)
            fallback = self.fallback
            return GroundedReasoningProtocolArtifact(
                protocol_markdown=fallback.strip(),
                techniques=["zero_shot_strict"],
                metadata={"warning": "ontology_missing"},
            )

        # 1) Prepare compact ontology for the synthesizer (token-safe)
        ontology_compact = compress_ontology_for_selection(
            ontology,
            max_items=getattr(self.cfg, "max_items", 200),
            max_string_chars=getattr(self.cfg, "max_string_chars", 5000),
            include_all_empty_ontology_keys=True,
        )

        # 2) Synthesize one or multiple candidate protocols
        system_prompt = self._build_system_prompt_reasoning_protocol()
        token_budget = int(getattr(self.cfg, "protocol_max_tokens", 4000) or 4000)
        user_prompt = self._build_user_prompt_reasoning_protocol(
            question=question.strip(),
            ontology_compact=ontology_compact,
            token_budget=token_budget,
        )

        token_count = count_tokens(system_prompt) + count_tokens(user_prompt)
        logger.info(f"[ReasoningProtocolDesigner] Prompt token count: {token_count}")

        model = getattr(
            self.cfg, "protocol_model", LLMVertexAIGeminiModels.GEMINI_3_0_FLASH
        )
        temperature = float(getattr(self.cfg, "temperature", 0.0) or 0.0)
        attempts = int(getattr(self.cfg, "max_attempts", 2) or 2)
        n = int(getattr(self.cfg, "protocol_self_consistency_n", 1) or 1)
        n = max(1, min(n, 5))

        candidates: List[GroundedReasoningProtocolArtifact] = []

        async def _single_candidate(
            run_idx: int,
        ) -> Optional[GroundedReasoningProtocolArtifact]:
            last_raw = ""
            for attempt in range(attempts):
                raw = await llmus.get_llm_response_with_backoff(
                    model=model,
                    system_prompt=system_prompt,
                    user_prompt=user_prompt
                    if attempt == 0
                    else self._repair_user_prompt(user_prompt, last_raw),
                    temperature=temperature,
                    caching=True,
                    max_tokens=token_budget,
                    request_timeout=self.cfg.agent_request_time_out,
                    fallback_models=self.cfg.fallback_llm_models,
                )
                last_raw = raw or ""
                payload = _extract_json_object(last_raw)
                if not payload:
                    continue
                art = self._validate_reasoning_protocol_payload(payload)
                if art and art.protocol_markdown.strip():
                    art.metadata = {
                        **(art.metadata or {}),
                        "candidate_run_idx": run_idx,
                        "attempt": attempt + 1,
                    }
                    return art
            return None

        if n == 1:
            art = await _single_candidate(0)
            if art:
                candidates.append(art)
        else:
            tasks = [_single_candidate(i) for i in range(n)]
            results = await asyncio.gather(*tasks, return_exceptions=True)
            for r in results:
                if (
                    isinstance(r, GroundedReasoningProtocolArtifact)
                    and r.protocol_markdown.strip()
                ):
                    candidates.append(r)

        if not candidates:
            fallback = self.fallback
            return GroundedReasoningProtocolArtifact(
                protocol_markdown=fallback.strip(),
                techniques=["zero_shot_strict"],
                metadata={"warning": "protocol_generation_failed"},
            )

        # If only one candidate, return it
        if len(candidates) == 1:
            return candidates[0]

        # 3) Judge: pick best candidate (cheap deterministic)
        judge_system = self._build_system_prompt_protocol_judge()
        judge_user = self._build_user_prompt_protocol_judge(
            question=question.strip(),
            candidates=[c.protocol_markdown for c in candidates],
        )

        judge_raw = await llmus.get_llm_response_with_backoff(
            model=model,
            system_prompt=judge_system,
            user_prompt=judge_user,
            temperature=0.0,
            caching=True,
            max_tokens=1000,
            request_timeout=self.cfg.agent_request_time_out,
            fallback_models=self.cfg.fallback_llm_models,
        )
        judge_payload = _extract_json_object((judge_raw or "").strip()) or {}
        best_index = judge_payload.get("best_index")

        if isinstance(best_index, int) and 0 <= best_index < len(candidates):
            picked = candidates[best_index]
            picked.metadata = {
                **(picked.metadata or {}),
                "self_consistency": True,
                "judge_why": str(judge_payload.get("why", ""))[:500],
                "candidates": len(candidates),
            }
            # Ensure techniques mention self-consistency if enabled
            if picked.techniques is None:
                picked.techniques = ["self_consistency_light"]
            elif "self_consistency_light" not in picked.techniques:
                picked.techniques.append("self_consistency_light")
            return picked

        # If judge fails, return the shortest “good” one (token-efficient default)
        picked = sorted(candidates, key=lambda a: len(a.protocol_markdown))[0]
        picked.metadata = {
            **(picked.metadata or {}),
            "self_consistency": True,
            "judge_failed": True,
        }
        if picked.techniques is None:
            picked.techniques = ["self_consistency_light"]
        return picked

    @staticmethod
    def _repair_user_prompt(original_user_prompt: str, last_raw: str) -> str:
        return json.dumps(
            {
                "repair_note": "Previous output was invalid. Return ONLY valid JSON matching schema.",
                "previous_output": (last_raw or "")[:2000],
                "original_request": original_user_prompt[:2000],
            },
            ensure_ascii=False,
        )

def _safe_str(x: Any) -> str:
    try:
        return str(x)
    except Exception as e:
        logger.warning(f"error {e}")
        return ""


def _truncate_str(s: str, max_chars: int) -> str:
    s = s.strip()
    if len(s) <= max_chars:
        return s
    return s[:max_chars].rstrip() + "…"


def _to_jsonable(obj: Any, max_string_chars: int) -> Any:
    if obj is None or isinstance(obj, (str, int, float, bool)):
        return _truncate_str(obj, max_string_chars) if isinstance(obj, str) else obj

    # pydantic v2
    if hasattr(obj, "model_dump"):
        try:
            return _to_jsonable(obj.model_dump(mode="python"), max_string_chars)
        except Exception as e:
            logger.warning(f"Error {e}")
            return _truncate_str(_safe_str(obj), max_string_chars)

    # pydantic v1
    if hasattr(obj, "dict"):
        try:
            return _to_jsonable(obj.dict(), max_string_chars)
        except Exception as e:
            logger.warning(f"Error {e}")
            return _truncate_str(_safe_str(obj), max_string_chars)

    if hasattr(obj, "tolist") and callable(getattr(obj, "tolist")):
        try:
            return _to_jsonable(obj.tolist(), max_string_chars)
        except Exception as e:
            logger.warning(f"Error {e}")
            return _truncate_str(_safe_str(obj), max_string_chars)

    if hasattr(obj, "item") and callable(getattr(obj, "item")):
        try:
            return _to_jsonable(obj.item(), max_string_chars)
        except Exception as e:
            logger.warning(f"Error {e}")
            return _truncate_str(_safe_str(obj), max_string_chars)

    if hasattr(obj, "isoformat") and callable(getattr(obj, "isoformat")):
        try:
            return obj.isoformat()
        except Exception as e:
            logger.warning(f"Error {e}")
            return _truncate_str(_safe_str(obj), max_string_chars)

    # containers
    if isinstance(obj, dict):
        return {str(k): _to_jsonable(v, max_string_chars) for k, v in obj.items()}
    if isinstance(obj, set):
        return [_to_jsonable(x, max_string_chars) for x in sorted(obj, key=_safe_str)]
    if isinstance(obj, (list, tuple)):
        return [_to_jsonable(x, max_string_chars) for x in obj]

    return _truncate_str(_safe_str(obj), max_string_chars)


def compress_ontology_for_selection(
    ontology: Dict[str, Any],
    *,
    max_items: int = 200,
    max_string_chars: int = 5000,
    include_all_empty_ontology_keys: bool = True,
) -> Dict[str, Dict[str, Any]]:
    """
    Compact ontology for Level-1 selection while preserving original ontology field names.

    Input ontology shape:
      {
        "<id>": {
          "graph": {...},
          "structure": {...},
          "rules": {...},
          "metadata": {...},
          "title": "...",
          "nlp_methodology_description": "...",
          ... possibly more fields ...
        },
        ...
      }

    Output (compact) shape:
      {
        "<id>": {
          # preserves original keys where possible, but trimmed
          "title": "...",
          "nlp_methodology_description": "...(truncated)...",
          "metadata": {
              "__keys__": ["k1","k2",...]
              # (optional future) "__sample__": {"k1": "...", ...}
          },
          # keep these keys as {} (optional) so schema stays recognizable
          "graph": {},
          "structure": {},
          "rules": {},
          # plus any additional unknown keys, truncated safely if strings
        },
        ...
      }
    """
    if not isinstance(ontology, dict) or not ontology:
        return {}

    compact: Dict[str, Dict[str, Any]] = {}

    for i, (o_id, o_value) in enumerate(ontology.items()):
        if i >= max_items:
            break

        if not isinstance(o_value, dict):
            compact[str(o_id)] = {"value": _to_jsonable(o_value, max_string_chars)}
            continue

        out_item: Dict[str, Any] = {}

        for k, v in o_value.items():
            out_item[k] = _to_jsonable(v, max_string_chars)

        if include_all_empty_ontology_keys:
            out_item.setdefault("graph", {})
            out_item.setdefault("structure", {})
            out_item.setdefault("rules", {})

        compact[str(o_id)] = out_item

    return compact

_JSON_OBJ_RE = re.compile(r"\{[\s\S]*\}")


def _extract_json_object(text: str) -> Optional[dict]:
    if not text:
        return None
    text = text.strip()
    if text.startswith("{") and text.endswith("}"):
        try:
            return json.loads(text)
        except Exception as e:
            logger.warning(f"JSON parse failed: {e}")
            pass
    m = _JSON_OBJ_RE.search(text)
    if not m:
        return None
    try:
        return json.loads(m.group(0))
    except Exception as e:
        logger.warning(f"JSON parse failed: {e}")
        return None


def _validate_ids(payload: dict, allowed_set: set[str]) -> List[str]:
    ids = payload.get("ids", [])
    if not isinstance(ids, list):
        return []
    out: List[str] = []
    for x in ids:
        if isinstance(x, str) and x in allowed_set:
            out.append(x)
    seen = set()
    dedup = []
    for x in out:
        if x not in seen:
            seen.add(x)
            dedup.append(x)
    return dedup


def _validate_fields(payload: dict, allowed_set: set[str]) -> List[str]:
    fields = payload.get("fields", [])
    if not isinstance(fields, list):
        return []
    out = []
    for f in fields:
        if isinstance(f, str) and f in allowed_set:
            out.append(f)
    # dedup stable
    seen = set()
    dedup = []
    for f in out:
        if f not in seen:
            seen.add(f)
            dedup.append(f)
    return dedup


def _validate_record_plan(payload: dict, allowed_set: set[str]) -> RecordPlan:
    rp = payload.get("record_plan")
    if not isinstance(rp, dict):
        return RecordPlan(mode="none")

    mode = rp.get("mode", "none")
    if mode not in RECORD_MODES:
        return RecordPlan(mode="none")

    plan = RecordPlan(mode=mode)

    # top_k
    if mode == "top_k":
        k = rp.get("k")
        sort_by = rp.get("sort_by")
        if not (isinstance(k, int) and 1 <= k <= 200):
            return RecordPlan(mode="none")
        if not (isinstance(sort_by, str) and sort_by in allowed_set):
            return RecordPlan(mode="none")
        plan.k = k
        plan.sort_by = sort_by
        plan.descending = bool(rp.get("descending", True))
        max_rows = rp.get("max_rows")
        if isinstance(max_rows, int) and 1 <= max_rows <= 1000:
            plan.max_rows = max_rows
        return plan

    # filter
    if mode == "filter":
        filters = rp.get("filters", [])
        if not isinstance(filters, list) or len(filters) == 0:
            return RecordPlan(mode="none")
        parsed: List[RecordFilter] = []
        for f in filters[:20]:
            if not isinstance(f, dict):
                continue
            field = f.get("field")
            op = f.get("op")
            value = f.get("value")
            if isinstance(field, str) and field in allowed_set and op in FILTER_OPS:
                parsed.append(RecordFilter(field=field, op=op, value=value))
        if not parsed:
            return RecordPlan(mode="none")
        plan.filters = parsed
        return plan

    # aggregate
    if mode == "aggregate":
        group_by = rp.get("group_by", [])
        aggregations = rp.get("aggregations", [])

        if not isinstance(aggregations, list) or len(aggregations) == 0:
            return RecordPlan(mode="none")

        # Validate group_by fields , i think we have to make sure field to group should be in allowed_set
        valid_group_by = []
        if isinstance(group_by, list):
            for g in group_by:
                if isinstance(g, str) and g in allowed_set:
                    valid_group_by.append(g)

        # Validate aggregations
        valid_aggs: List[Aggregation] = []
        for agg in aggregations[:20]:
            if not isinstance(agg, dict):
                continue
            agg_field = agg.get("field")
            agg_op = agg.get("op")
            agg_alias = agg.get("alias")
            if (
                isinstance(agg_field, str)
                and agg_field in allowed_set
                and agg_op in AGGREGATE_OPS
            ):
                valid_aggs.append(
                    Aggregation(field=agg_field, op=agg_op, alias=agg_alias)
                )

        if not valid_aggs:
            return RecordPlan(mode="none")

        plan.group_by = valid_group_by if valid_group_by else None
        plan.aggregations = valid_aggs
        # Also support filters in aggregate mode, that I think we can replace the filter, but filter for Op Agent means we get list after filter, here we get aggregation result after filter, tradeoff need to be consider.
        filters = rp.get("filters", [])
        if isinstance(filters, list) and filters:
            parsed_filters: List[RecordFilter] = []
            # I think I support maximum 20 filters, I don't think we should support more than that
            for f in filters[:20]:
                if not isinstance(f, dict):
                    continue
                field = f.get("field")
                op = f.get("op")
                value = f.get("value")
                # TODO: Operation list NEED to be dynamic
                if isinstance(field, str) and field in allowed_set and op in FILTER_OPS:
                    parsed_filters.append(RecordFilter(field=field, op=op, value=value))
            if parsed_filters:
                plan.filters = parsed_filters
        return plan

    # all/none
    if mode == "all":
        max_rows = rp.get("max_rows")
        if isinstance(max_rows, int) and 1 <= max_rows <= 1000:
            plan.max_rows = max_rows
    return plan


def _validate_section_plan_payload(
    payload: dict,
    *,
    allowed_fields_by_sid: Dict[str, List[str]],
) -> Dict[str, Tuple[List[str], RecordPlan]]:
    """
    payload schema:
      {
        "sections": {
          "<sid>": {
            "fields": [...],
            "record_plan": {...}
          },
          ...
        }
      }
    """
    sections = payload.get("sections", {})
    if not isinstance(sections, dict):
        return {}

    out: Dict[str, Tuple[List[str], RecordPlan]] = {}

    for sid, plan in sections.items():
        if sid not in allowed_fields_by_sid:
            continue
        if not isinstance(plan, dict):
            continue

        allowed_set = set(allowed_fields_by_sid[sid])

        fields = _validate_fields({"fields": plan.get("fields", [])}, allowed_set)
        rp = _validate_record_plan(
            {"record_plan": plan.get("record_plan", {})}, allowed_set
        )

        out[sid] = (fields, rp)

    return out


def _should_use_mode_all(
    sid: str,
    row_count: int,
    important_partitions: Optional[List[str]] = None,
) -> bool:
    """
    Determine if a section should use mode='all' based on heuristics.
    Returns True if: starts with 'summarize_' OR row_count <= 50 OR in important_partitions.
    """
    if sid.startswith("summarize_"):
        return True
    if row_count <= 50:
        return True
    if (
        important_partitions is not None
        and len(important_partitions) > 0
        and sid in important_partitions
    ):
        return True
    return False

@dataclass
class QAOntologyDrivenContentSelectorConfig:
    # Candidate selection hyperparameters
    max_items: int = 200
    max_meta_keys: int = 100
    max_string_chars: int = 5000
    max_fields_per_section: int = 100
    max_rows_for_mode_all: int = 500

    # Internal processing hyperparameters
    max_concurrency: int = 10
    enable_aggregate_mode: bool = False

    # LLM hyperparameters
    temperature: float = 0.0
    max_output_tokens: int = 100_000
    max_input_tokens: int = 100_000
    max_attempts: int = 2
    agent_request_time_out: int = 60
    # No client data sending here ( just ontology and question) ==> we can use GPTs or Claude
    main_llm_model: str = DEFAULT_SEMANTIC_QUERY_PLANNER_LLM_MODEL
    fallback_llm_models: List[str] = field(
        default_factory=lambda: DEFAULT_SEMANTIC_QUERY_PLANNER_MODEL_FALLBACKS
    )

    @classmethod
    def from_query_planner_config(
        cls,
        config: Optional[SemanticQueryPlannerConfig] = None,
        enable_aggregate_mode: bool = False,
    ) -> "QAOntologyDrivenContentSelectorConfig":
        if config is None:
            return cls(enable_aggregate_mode=enable_aggregate_mode)

        return cls(
            temperature=config.temperature,
            max_output_tokens=config.max_output_tokens,
            max_input_tokens=config.max_input_tokens,
            max_attempts=config.max_attempts,
            agent_request_time_out=config.agent_request_time_out,
            main_llm_model=config.main_llm_model
            or DEFAULT_SEMANTIC_QUERY_PLANNER_LLM_MODEL,
            fallback_llm_models=config.fallback_llm_models
            or DEFAULT_SEMANTIC_QUERY_PLANNER_MODEL_FALLBACKS,
            enable_aggregate_mode=enable_aggregate_mode,
        )

    async def select_ontology_content_ids(
        self,
        *,
        question: str,
        ontology: Dict[str, Any],
        suggested_selected_partitions: Optional[List[str]] = None,
        use_suggestions_only: bool = False,
    ) -> List[str]:
        if not question or not question.strip():
            return []
        if not ontology:
            return []

        # Skip normal logic since we want exactly the designated suggestions
        if (
            use_suggestions_only
            and suggested_selected_partitions
            and len(suggested_selected_partitions) > 0
        ):
            allowed_set = set(ontology.keys())
            logger.info(
                "[Q&A Semantic Query Planner] Ignore to use Smart Partition Selection as workflow output sections might be small"
            )
            return [sid for sid in suggested_selected_partitions if sid in allowed_set]

        allowed_ids = list(ontology.keys())
        allowed_set = set(allowed_ids)
        compact = compress_ontology_for_selection(
            ontology,
            max_items=self.cfg.max_items,
            max_string_chars=self.cfg.max_string_chars,
            include_all_empty_ontology_keys=True,
        )
        user_prompt = self._build_user_prompt_select_ids(
            question=question.strip(),
            ontology_compact=compact,
            allowed_ids=allowed_ids,
        )
        system_prompt = self._build_system_prompt_select_ids()
        last_raw = ""
        for attempt in range(self.cfg.max_attempts):
            raw = await llmus.get_llm_response_with_backoff(
                model=self.cfg.main_llm_model,
                system_prompt=system_prompt,
                user_prompt=user_prompt
                if attempt == 0
                else self._repair_user_prompt(user_prompt, last_raw),
                temperature=self.cfg.temperature,
                caching=True,
                request_timeout=self.cfg.agent_request_time_out,
                fallback_models=self.cfg.fallback_llm_models,
            )
            last_raw = raw or ""
            payload = _extract_json_object(last_raw)
            if not payload:
                continue
            ids = _validate_ids(payload, allowed_set)
            logger.info(
                f"[Q&A Semantic Query Planner] Attempt {attempt + 1}: Smart Extracted section IDs: {ids}"
            )
            # Merge with suggested_selected_partitions
            if suggested_selected_partitions:
                suggests_valid = [
                    sid for sid in suggested_selected_partitions if sid in allowed_set
                ]
                seen = set(suggests_valid)
                completed_ids = suggests_valid + [sid for sid in ids if sid not in seen]
                logger.info(
                    f"[Q&A Semantic Query Planner] ---- Completed Extracted section IDs: {completed_ids}"
                )
                return completed_ids
            # Return always_select_ids even if LLM selection failed
            return ids
        if suggested_selected_partitions:
            return [sid for sid in suggested_selected_partitions if sid in allowed_set]
        return []

    async def select_fields_for_section(
        self,
        *,
        question: str,
        section_id: str,
        allowed_fields: List[str],
        section_ontology: Dict[str, Any],
    ) -> List[str]:
        if not question or not question.strip():
            return []
        if not section_id:
            return []
        if not allowed_fields:
            return []

        allowed_set = set(allowed_fields)
        system_prompt = self._build_system_prompt_select_fields()
        user_prompt = self._build_user_prompt_select_fields(
            question=question.strip(),
            section_id=section_id,
            allowed_fields=allowed_fields,
            section_ontology=section_ontology,
        )

        token_size = count_tokens(system_prompt) + count_tokens(user_prompt)
        logger.info(
            f"[Q&A Semantic Query Planner - Level-2 - Columns] Extracting fields for section {section_id}. Token size: {token_size}"
        )

        if token_size > self.cfg.max_input_tokens:
            logger.warning(
                f"[Q&A Semantic Query Planner - Level-2 - Columns] Token size {token_size} exceeds max_input_tokens {self.cfg.max_input_tokens}. Skipping field selection."
            )
            return []

        last_raw = ""
        for attempt in range(self.cfg.max_attempts):
            raw = await llmus.get_llm_response_with_backoff(
                model=self.cfg.main_llm_model,
                system_prompt=system_prompt,
                user_prompt=user_prompt
                if attempt == 0
                else self._repair_user_prompt(user_prompt, last_raw),
                temperature=self.cfg.temperature,
                caching=True,
                request_timeout=self.cfg.agent_request_time_out,
                fallback_models=self.cfg.fallback_llm_models,
            )
            last_raw = raw or ""
            payload = _extract_json_object(last_raw)
            if not payload:
                continue

            fields = _validate_fields(payload, allowed_set)
            if len(fields) > self.cfg.max_fields_per_section:
                logger.warning(
                    f"[Q&A Semantic Query Planner - Level-2 - Columns] section={section_id} capping fields from {len(fields)} to {self.cfg.max_fields_per_section}"
                )
                fields = fields[: self.cfg.max_fields_per_section]

            logger.info(
                f"[Q&A Semantic Query Planner - Level-2 - Columns] section={section_id} fields={fields}"
            )
            return fields

        return []

    async def select_record_plan_for_section(
        self,
        *,
        question: str,
        section_id: str,
        allowed_fields: List[str],
        selected_fields: List[str],
        section_ontology: Dict[str, Any],
        row_count: int,
    ) -> RecordPlan:
        if not question or not question.strip():
            return RecordPlan(mode="none")
        if not section_id:
            return RecordPlan(mode="none")
        if not allowed_fields:
            return RecordPlan(mode="none")

        allowed_set = set(allowed_fields)

        system_prompt = self._build_system_prompt_select_records()
        user_prompt = self._build_user_prompt_select_records(
            question=question.strip(),
            section_id=section_id,
            allowed_fields=allowed_fields,
            selected_fields=selected_fields or [],
            row_count=int(row_count or 0),
            section_ontology=section_ontology,
        )

        token_size = count_tokens(system_prompt) + count_tokens(user_prompt)
        logger.info(
            f"[Q&A Semantic Query Planner - Level-3 RecordPlan] Extracting record plan for section {section_id}. Token size: {token_size}"
        )

        if token_size > self.cfg.max_input_tokens:
            logger.warning(
                f"[Q&A Semantic Query Planner - Level-3 RecordPlan] Token size {token_size} exceeds max_input_tokens {self.cfg.max_input_tokens}. Skipping record plan selection."
            )
            return RecordPlan(mode="none")

        last_raw = ""
        for attempt in range(self.cfg.max_attempts):
            raw = await llmus.get_llm_response_with_backoff(
                model=self.cfg.main_llm_model,
                system_prompt=system_prompt,
                user_prompt=user_prompt
                if attempt == 0
                else self._repair_user_prompt(user_prompt, last_raw),
                temperature=self.cfg.temperature,
                caching=True,
                request_timeout=self.cfg.agent_request_time_out,
                fallback_models=self.cfg.fallback_llm_models,
            )
            last_raw = raw or ""
            payload = _extract_json_object(last_raw)
            if not payload:
                continue

            plan = _validate_record_plan(payload, allowed_set)
            logger.info(
                f"[Q&A Semantic Query Planner - Level-3 RecordPlan] section={section_id} plan={plan}"
            )
            return plan

        return RecordPlan(mode="none")

    async def select_fields_and_record_plans_batch_all_sections(
        self,
        *,
        question: str,
        section_ids: List[str],
        ontology: Dict[str, Any],
        structured_output_by_id: Dict[str, Any],
    ) -> Dict[str, Tuple[List[str], RecordPlan]]:
        if not question or not question.strip() or not section_ids:
            return {}

        # Build per-section inputs guarded by structured_output_by_id
        sections_payload: List[dict] = []
        allowed_fields_by_sid: Dict[str, List[str]] = {}

        ontology_subset = {sid: ontology[sid] for sid in section_ids if sid in ontology}
        ontology_compact = compress_ontology_for_selection(
            ontology_subset,
            max_items=len(ontology_subset),
            max_string_chars=self.cfg.max_string_chars,
            include_all_empty_ontology_keys=True,
        )

        for sid in section_ids:
            so = structured_output_by_id.get(sid)
            if not so:
                continue

            data = (
                getattr(so, "data", None)
                if not isinstance(so, dict)
                else so.get("data")
            )
            if not isinstance(data, list) or not data:
                continue

            headers = data[0] if isinstance(data[0], list) else []
            allowed_fields = [h for h in headers if isinstance(h, str)]
            if not allowed_fields:
                continue

            row_count = max(0, len(data) - 1)
            allowed_fields_by_sid[sid] = allowed_fields

            # ontology metadata desc (subset to allowed_fields)
            meta_raw = {}
            o_item = ontology.get(sid, {})
            if isinstance(o_item, dict):
                meta_raw = o_item.get("metadata", {})

            meta_desc: Dict[str, Any] = {}
            if isinstance(meta_raw, dict):
                meta_desc = {
                    k: meta_raw.get(k) for k in allowed_fields if k in meta_raw
                }

            o_item_compact = ontology_compact.get(sid, {})
            if not isinstance(o_item_compact, dict):
                o_item_compact = {}
            o_item_compact.setdefault("metadata", meta_desc)

            sections_payload.append(
                {
                    "id": sid,
                    "allowed_fields": allowed_fields,
                    "row_count": row_count,
                    "section_ontology": o_item_compact,
                }
            )

        if not sections_payload:
            return {}

        system_prompt = self._build_system_prompt_batch_fields_and_records()
        user_prompt = self._build_user_prompt_batch_fields_and_records(
            question=question.strip(),
            sections=sections_payload,
        )

        token_size = count_tokens(system_prompt) + count_tokens(user_prompt)
        logger.info(
            f"[Q&A Semantic Query Planner - Leve2+3 - BatchMode] Semantic Select FIELDS and RECORDS for sections. Batch fields and records: token_size={token_size}"
        )
        if token_size > self.cfg.max_input_tokens:
            logger.warning(
                f"[Q&A Semantic Query Planner - Leve2+3 - BatchMode] Token size {token_size} exceeds max_input_tokens {self.cfg.max_input_tokens}. Skipping batch field and record plan selection."
            )
            return {}

        last_raw = ""
        for attempt in range(self.cfg.max_attempts):
            raw = await llmus.get_llm_response_with_backoff(
                model=self.cfg.main_llm_model,
                system_prompt=system_prompt.strip(),
                user_prompt=user_prompt.strip()
                if attempt == 0
                else self._repair_user_prompt(user_prompt.strip(), last_raw),
                temperature=self.cfg.temperature,
                caching=True,
                request_timeout=self.cfg.agent_request_time_out,
                fallback_models=self.cfg.fallback_llm_models,
            )
            last_raw = raw or ""
            payload = _extract_json_object(last_raw)
            if not payload:
                continue
            fields_and_records = _validate_section_plan_payload(
                payload, allowed_fields_by_sid=allowed_fields_by_sid
            )
            return fields_and_records

        return {}

    # Real public API here
    async def build_semantic_query_plan_batch(
        self,
        *,
        question: str,
        ontology: Dict[str, Any],
        structured_output_by_id: Dict[str, Any],
        suggested_selected_partitions: Optional[List[str]] = None,
        use_suggestions_only: bool = False,
        important_partitions: Optional[List[str]] = None,
        enable_important_partitions_fast_path: bool = True,
    ) -> SemanticQueryPlan:
        section_ids = await self.select_ontology_content_ids(
            question=question,
            ontology=ontology,
            suggested_selected_partitions=suggested_selected_partitions,
            use_suggestions_only=use_suggestions_only,
        )

        # Separate important and regular sections
        important_set = set(important_partitions) if important_partitions else set()
        if enable_important_partitions_fast_path and important_set:
            important_section_ids = [sid for sid in section_ids if sid in important_set]
            regular_section_ids = [
                sid for sid in section_ids if sid not in important_set
            ]
        else:
            important_section_ids = []
            regular_section_ids = section_ids

        plans: List[SectionSelectionPlan] = []

        # FAST PATH: Important partitions only when enabled
        for sid in important_section_ids:
            so = structured_output_by_id.get(sid)
            if not so:
                continue
            data = (
                getattr(so, "data", None)
                if not isinstance(so, dict)
                else so.get("data")
            )
            if not isinstance(data, list) or not data:
                continue
            headers = data[0] if isinstance(data[0], list) else []
            allowed_fields = [h for h in headers if isinstance(h, str)]
            if not allowed_fields:
                continue

            fields = allowed_fields[: self.cfg.max_fields_per_section]
            record_plan = RecordPlan(mode="all")
            logger.info(
                f"[Q&A Semantic Query Planner] Important partition '{sid}': using all {len(fields)} fields, mode=all (#-row= {len(data) - 1})"
            )
            plans.append(
                SectionSelectionPlan(id=sid, fields=fields, record_plan=record_plan)
            )

        if regular_section_ids and len(regular_section_ids) > 0:
            row_count_by_sid: Dict[str, int] = {}
            for sid in regular_section_ids:
                so = structured_output_by_id.get(sid)
                if not so:
                    continue
                data = (
                    getattr(so, "data", None)
                    if not isinstance(so, dict)
                    else so.get("data")
                )
                if not isinstance(data, list) or not data:
                    continue
                headers = data[0] if isinstance(data[0], list) else []
                allowed_fields = [h for h in headers if isinstance(h, str)]
                if not allowed_fields:
                    continue
                row_count_by_sid[sid] = max(0, len(data) - 1)

                # Ontology drift validation
                o_item = ontology.get(sid, {})
                meta_raw = (
                    o_item.get("metadata", {}) if isinstance(o_item, dict) else {}
                )
                if not isinstance(meta_raw, dict) or not meta_raw:
                    logger.warning(
                        f"[Q&A Ontology Drift] section={sid} has no ontology.metadata (or not a dict). "
                        f"headers_count={len(allowed_fields)}"
                    )
                else:
                    headers_set = set(allowed_fields)
                    meta_keys_set = set(meta_raw.keys())
                    missing_in_ontology = sorted(headers_set - meta_keys_set)
                    extra_in_ontology = sorted(meta_keys_set - headers_set)
                    if missing_in_ontology:
                        logger.warning(
                            f"[Q&A Ontology Drift] section={sid} headers not found in ontology.metadata: "
                            f"{missing_in_ontology[:50]}{' ...' if len(missing_in_ontology) > 50 else ''}"
                        )
                    if extra_in_ontology:
                        logger.warning(
                            f"[Q&A Ontology Drift] section={sid} ontology.metadata keys not present in table headers: "
                            f"{extra_in_ontology[:50]}{' ...' if len(extra_in_ontology) > 50 else ''}"
                        )

            plan_map = await self.select_fields_and_record_plans_batch_all_sections(
                question=question,
                section_ids=regular_section_ids,
                ontology=ontology,
                structured_output_by_id=structured_output_by_id,
            )

            for sid in regular_section_ids:
                fields, record_plan = plan_map.get(sid, ([], RecordPlan(mode="none")))
                row_count = row_count_by_sid.get(sid, 0)

                # Override record_plan with heuristic if applicable
                # Pass important_partitions when fast path is disabled to force mode="all"
                effective_important = (
                    important_partitions
                    if not enable_important_partitions_fast_path
                    else None
                )
                if fields:
                    if _should_use_mode_all(
                        sid, row_count, important_partitions=effective_important
                    ):
                        record_plan = RecordPlan(mode="all")
                    elif row_count <= 100:
                        record_plan = RecordPlan(mode="all", max_rows=row_count)
                    elif record_plan.mode == "none":
                        # Fallback if LLM returned mode="none" for important partition
                        record_plan = RecordPlan(
                            mode="top_k",
                            k=30,
                            sort_by=fields[0] if fields else None,
                        )
                logger.info(
                    f"[Q&A Semantic Query Planner - Leve2+3 - BatchMode] section={sid} ; fields={fields} ; record_plan={record_plan}"
                )
                plans.append(
                    SectionSelectionPlan(id=sid, fields=fields, record_plan=record_plan)
                )
        return SemanticQueryPlan(sections=plans)

    async def build_semantic_query_plan(
        self,
        *,
        question: str,
        ontology: Dict[str, Any],
        structured_output_by_id: Dict[str, Any],
        suggested_selected_partitions: Optional[List[str]] = None,
        use_suggestions_only: bool = False,
        important_partitions: Optional[List[str]] = None,
        enable_important_partitions_fast_path: bool = True,
    ) -> SemanticQueryPlan:
        """
        Level-1 + Level-2 + Level-3.
        - Level-1: section ids from ontology
        - Level-2: fields from section headers + ontology.metadata
        - Level-3: record plan (all/top_k/filter/none)
        """
        section_ids = await self.select_ontology_content_ids(
            question=question,
            ontology=ontology,
            suggested_selected_partitions=suggested_selected_partitions,
            use_suggestions_only=use_suggestions_only,
        )

        ontology_subset = {
            sid: ontology.get(sid) for sid in section_ids if sid in ontology
        }
        ontology_compact = compress_ontology_for_selection(
            ontology_subset,
            max_items=len(ontology_subset),
            max_string_chars=self.cfg.max_string_chars,
            include_all_empty_ontology_keys=True,
        )

        sem = asyncio.Semaphore(self.cfg.max_concurrency)

        async def _analysis_single_section(
            idx: int, sid: str
        ) -> Tuple[int, Optional[SectionSelectionPlan]]:
            async with sem:
                so = structured_output_by_id.get(sid)
                if not so:
                    return idx, None

                data = (
                    getattr(so, "data", None)
                    if not isinstance(so, dict)
                    else so.get("data")
                )
                if not isinstance(data, list) or not data:
                    return idx, None

                headers = data[0] if isinstance(data[0], list) else []
                allowed_fields = [h for h in headers if isinstance(h, str)]
                if not allowed_fields:
                    return idx, None

                # FAST PATH: Important partitions - skip LLM, get all fields and all rows
                if (
                    enable_important_partitions_fast_path
                    and important_partitions
                    and sid in important_partitions
                ):
                    fields = allowed_fields[: self.cfg.max_fields_per_section]
                    record_plan = RecordPlan(mode="all")
                    logger.info(
                        f"[Q&A Semantic Query Planner] Important partition '{sid}': using all {len(fields)} fields, mode=all"
                    )
                    return idx, SectionSelectionPlan(
                        id=sid, fields=fields, record_plan=record_plan
                    )

                # Validation
                o_item_raw = ontology.get(sid, {})
                meta_raw = (
                    o_item_raw.get("metadata", {})
                    if isinstance(o_item_raw, dict)
                    else {}
                )

                if not isinstance(meta_raw, dict) or not meta_raw:
                    logger.warning(
                        f"[Q&A Ontology Drift] section={sid} has no ontology.metadata (or not a dict). "
                        f"headers_count={len(allowed_fields)}"
                    )
                else:
                    headers_set = set(allowed_fields)
                    meta_keys_set = set(meta_raw.keys())
                    missing_in_ontology = sorted(headers_set - meta_keys_set)
                    extra_in_ontology = sorted(meta_keys_set - headers_set)

                    if missing_in_ontology:
                        logger.warning(
                            f"[Q&A Ontology Drift] section={sid} headers not found in ontology.metadata: "
                            f"{missing_in_ontology[:50]}{' ...' if len(missing_in_ontology) > 50 else ''}"
                        )
                    if extra_in_ontology:
                        logger.warning(
                            f"[Q&A Ontology Drift] section={sid} ontology.metadata keys not present in table headers: "
                            f"{extra_in_ontology[:50]}{' ...' if len(extra_in_ontology) > 50 else ''}"
                        )

                # Section ontology context
                meta_desc: Dict[str, Any] = {}
                if isinstance(meta_raw, dict):
                    meta_desc = {
                        k: meta_raw.get(k) for k in allowed_fields if k in meta_raw
                    }

                o_item_compact = ontology_compact.get(sid, {})
                if not isinstance(o_item_compact, dict):
                    o_item_compact = {}
                o_item_compact.setdefault("metadata", meta_desc)

                fields = await self.select_fields_for_section(
                    question=question,
                    section_id=sid,
                    allowed_fields=allowed_fields,
                    section_ontology=o_item_compact,
                )

                if not fields:
                    return idx, SectionSelectionPlan(
                        id=sid,
                        fields=[],
                        record_plan=RecordPlan(mode="none"),
                    )

                row_count = max(0, len(data) - 1)

                # small heuristic: summary tables usually should be mode=all == Important to keep the most important fields, and avoid over-filtering. For larger tables, we can be more aggressive.
                if _should_use_mode_all(sid, row_count, important_partitions):
                    record_plan = RecordPlan(mode="all")
                elif row_count <= 100:
                    record_plan = RecordPlan(mode="all", max_rows=row_count)
                else:
                    record_plan = await self.select_record_plan_for_section(
                        question=question,
                        section_id=sid,
                        allowed_fields=allowed_fields,
                        selected_fields=fields,
                        row_count=row_count,
                        section_ontology=o_item_compact,
                    )

                    if record_plan.mode == "none":
                        record_plan = RecordPlan(
                            mode="top_k",
                            k=30,
                            sort_by=fields[0] if fields else allowed_fields[0],
                        )

                return idx, SectionSelectionPlan(
                    id=sid, fields=fields, record_plan=record_plan
                )

        # Fire tasks for all sections
        tasks = [
            asyncio.create_task(_analysis_single_section(i, sid))
            for i, sid in enumerate(section_ids)
        ]
        results = await asyncio.gather(*tasks)

        # Preserve original section_ids order
        results.sort(key=lambda x: x[0])
        plans = [plan for _, plan in results if plan is not None]

        return SemanticQueryPlan(sections=plans)

    @staticmethod
    def _repair_user_prompt(original_user_prompt: str, last_raw: str) -> str:
        try:
            original = json.loads(original_user_prompt)
        except Exception:
            original = {"raw": original_user_prompt[:2000]}

        repair = {
            "repair_note": "Your previous output was invalid. Return ONLY valid JSON matching schema.",
            "previous_output": (last_raw or "")[:100000],
            "original_request": original,
        }
        return json.dumps(repair, ensure_ascii=False)

class BaseSemanticQueryPlanner(ABC):
    @abstractmethod
    def get_plan_function(
        self, use_batch_mode: bool
    ) -> Callable[..., Awaitable[SemanticQueryPlan]]:
        """
        Get the plan function based on batch mode.

        Args:
           use_batch_mode: True for batch mode, False for parallel mode

        Returns:
           Async function that generates SemanticQueryPlan
        """
        pass


class LLMOntologyDrivenQueryPlanner(BaseSemanticQueryPlanner):
    """LLM-based and Ontology-Driven query planner."""

    def __init__(
        self,
        config: SemanticQueryPlannerConfig,
        enable_aggregate_mode: bool = False,
    ):
        self.config = config
        self.selector = QAOntologyDrivenContentSelector(
            semantic_query_planner_config=config,
            enable_aggregate_mode=enable_aggregate_mode,
        )

    def get_plan_function(
        self, use_batch_mode: bool
    ) -> Callable[..., Awaitable[SemanticQueryPlan]]:
        """Return batch or parallel plan function."""
        if use_batch_mode:
            return self.selector.build_semantic_query_plan_batch
        return self.selector.build_semantic_query_plan

    def get_plan_function(
        self, use_batch_mode: bool
    ) -> Callable[..., Awaitable[SemanticQueryPlan]]:

def create_semantic_query_planner(
    config: SemanticQueryPlannerConfig,
    enable_aggregate_mode: bool = False,
    important_partitions: Optional[List[str]] = None,
    enable_important_partitions_fast_path: bool = True,
) -> BaseSemanticQueryPlanner:
    """Factory to create semantic query planner based on strategy."""
    strategy = config.strategy

    if strategy == SemanticQueryPlannerStrategy.LLM_ONTOLOGY_DRIVEN:
        return LLMOntologyDrivenQueryPlanner(
            config=config,
            enable_aggregate_mode=enable_aggregate_mode,
        )

    elif strategy == SemanticQueryPlannerStrategy.DETERMINISTIC:
        return DeterministicQueryPlannerWrapper(
            config=config,
            enable_aggregate_mode=enable_aggregate_mode,
            important_partitions=important_partitions,
            enable_important_partitions_fast_path=enable_important_partitions_fast_path,
        )

    elif strategy == SemanticQueryPlannerStrategy.HYBRID:
        return HybridQueryPlanner(
            config=config,
            enable_aggregate_mode=enable_aggregate_mode,
            important_partitions=important_partitions,
            enable_important_partitions_fast_path=enable_important_partitions_fast_path,
        )

    else:
        raise ValueError(f"Unknown query planner strategy: {strategy}")

def _is_number(x: Any) -> bool:
    return isinstance(x, (int, float)) and not isinstance(x, bool)


def _to_float_or_none(x: Any) -> Optional[float]:
    if x is None:
        return None
    if _is_number(x):
        return float(x)
    if isinstance(x, str):
        s = x.strip().replace(",", "")
        if s == "":
            return None
        if s.endswith("%"):
            try:
                return float(s[:-1].strip())
            except Exception:
                return None
        try:
            return float(s)
        except Exception:
            return None
    return None


def _normalize_str(x: Any) -> str:
    return "" if x is None else str(x)


def _build_header_index(header_row: List[Any]) -> Dict[str, int]:
    idx: Dict[str, int] = {}
    for i, h in enumerate(header_row):
        if isinstance(h, str) and h.strip():
            idx[h] = i
    return idx


def _project_table_to_fields(
    data: List[List[Any]], fields: List[str]
) -> List[List[Any]]:
    if not data or not isinstance(data, list) or not isinstance(data[0], list):
        return data

    header = data[0]
    header_idx = _build_header_index(header)

    kept_fields: List[str] = []
    kept_indices: List[int] = []
    for f in fields:
        if f in header_idx:
            kept_fields.append(f)
            kept_indices.append(header_idx[f])
        else:
            logger.warning(
                f"[Q&A Smart Retriever] Requested field not in headers: field={f}"
            )

    if not kept_indices:
        return [header]

    out: List[List[Any]] = [kept_fields]
    for row in data[1:]:
        if not isinstance(row, list):
            continue
        out.append([row[i] if i < len(row) else "" for i in kept_indices])
    return out


def _eval_filter_op(cell: Any, op: str, value: Any) -> bool:
    """
    Supported ops in Q&A models
    TODO: We should extend the filtering operation as many as we can
    Notes:
    - 'contains' does case-insensitive substring on string forms
    - 'in' supports list/tuple/set membership (string membership also supported)
    - numeric comparisons attempt float coercion
    """
    if op == "==":
        return cell == value
    if op == "!=":
        return cell != value

    if op == "contains":
        a = _normalize_str(cell).lower()
        b = _normalize_str(value).lower()
        return b in a

    if op == "in":
        if isinstance(value, (list, tuple, set)):
            return cell in value or _normalize_str(cell) in {str(v) for v in value}
        return _normalize_str(value).lower() in _normalize_str(cell).lower()

    a_num = _to_float_or_none(cell)
    b_num = _to_float_or_none(value)
    if a_num is None or b_num is None:
        return False

    if op == ">":
        return a_num > b_num
    if op == ">=":
        return a_num >= b_num
    if op == "<":
        return a_num < b_num
    if op == "<=":
        return a_num <= b_num

    return False


def _apply_filters(data: List[List[Any]], filters: List[Any]) -> List[List[Any]]:
    if not data or len(data) <= 1:
        return data
    header = data[0]
    header_idx = _build_header_index(header)
    rows = data[1:]

    # compile usable filters
    compiled: List[Tuple[int, str, Any]] = []
    for f in filters:
        try:
            field = (
                getattr(f, "field", None) if not isinstance(f, dict) else f.get("field")
            )
            op = getattr(f, "op", None) if not isinstance(f, dict) else f.get("op")
            val = (
                getattr(f, "value", None) if not isinstance(f, dict) else f.get("value")
            )
            if isinstance(field, str) and field in header_idx and isinstance(op, str):
                compiled.append((header_idx[field], op, val))
            else:
                logger.warning(f"[Q&A Smart Retriever] Skipping invalid filter: {f}")
        except Exception as e:
            logger.warning(
                f"[Q&A Smart Retriever] Skipping invalid filter (exception): {e}"
            )

    if not compiled:
        return data

    out_rows: List[List[Any]] = []
    for r in rows:
        if not isinstance(r, list):
            continue
        ok = True
        for col_idx, op, val in compiled:
            cell = r[col_idx] if col_idx < len(r) else ""
            if not _eval_filter_op(cell, op, val):
                ok = False
                break
        if ok:
            out_rows.append(r)

    return [header] + out_rows


def _apply_top_k(
    data: List[List[Any]],
    *,
    k: int,
    sort_by: str,
    descending: bool = True,
) -> List[List[Any]]:
    if not data or len(data) <= 1:
        return data
    header = data[0]
    header_idx = _build_header_index(header)
    if sort_by not in header_idx:
        logger.warning(
            f"[Q&A Smart Retriever] top_k sort_by not found in headers: sort_by={sort_by}"
        )
        return data

    sort_col = header_idx[sort_by]
    rows = [r for r in data[1:] if isinstance(r, list)]

    def sort_key(row: List[Any]) -> Tuple[int, float, str]:
        cell = row[sort_col] if sort_col < len(row) else ""
        num = _to_float_or_none(cell)
        if num is None:
            # marker 1 means "non-numeric"
            return (1, 0.0, _normalize_str(cell))
        return (0, num, "")

    rows_sorted = sorted(rows, key=sort_key, reverse=bool(descending))
    k = max(1, min(int(k), len(rows_sorted)))
    return [header] + rows_sorted[:k]


def _data_to_dataframe(data: List[List[Any]]) -> Optional[pd.DataFrame]:
    """Convert list-of-lists data to DataFrame."""
    if not data or len(data) <= 1:
        return None
    header = data[0]
    rows = data[1:]
    try:
        df = pd.DataFrame(rows, columns=header)
        return df
    except Exception as e:
        logger.warning(f"[Q&A Smart Retriever] Failed to create DataFrame: {e}")
        return None


def _dataframe_to_data(df: pd.DataFrame) -> List[List[Any]]:
    """Convert DataFrame back to list-of-lists."""
    if df is None or df.empty:
        return []
    header = df.columns.tolist()
    rows = df.values.tolist()
    return [header] + rows


def _apply_single_filter(
    df: pd.DataFrame,
    field: str,
    op: str,
    value: Any,
) -> pd.DataFrame:
    """Apply a single filter operation to DataFrame."""
    if field not in df.columns:
        logger.warning(f"[Q&A Smart Retriever] Filter field not found: {field}")
        return df

    col = df[field]

    # String operations - case-insensitive
    if op in STRING_COMPARISON_OPS:
        # Convert column to string, lowercase for comparison
        col_str = col.astype(str).str.lower().str.strip()
        value_str = str(value).lower().strip()

        if op == "contains":
            mask = col_str.str.contains(value_str, na=False, regex=False)
        elif op == "==":
            # Exact match (case-insensitive for strings)
            mask = col_str == value_str
        elif op == "!=":
            mask = col_str != value_str
        elif op == "in":
            if isinstance(value, (list, tuple, set)):
                value_list = [str(v).lower().strip() for v in value]
                mask = col_str.isin(value_list)
            else:
                mask = col_str.str.contains(value_str, na=False, regex=False)
        elif op == "startswith":
            mask = col_str.str.startswith(value_str, na=False)
        elif op == "endswith":
            mask = col_str.str.endswith(value_str, na=False)
        else:
            mask = pd.Series([True] * len(df))

    # Numeric operations
    elif op in NUMERICAL_COMPARISON_OPS:
        col_numeric = pd.to_numeric(col, errors="coerce")
        value_numeric = pd.to_numeric(value, errors="coerce")

        if pd.isna(value_numeric):
            logger.warning(f"[Q&A Smart Retriever] Filter value not numeric: {value}")
            return df

        if op == ">":
            mask = col_numeric > value_numeric
        elif op == ">=":
            mask = col_numeric >= value_numeric
        elif op == "<":
            mask = col_numeric < value_numeric
        elif op == "<=":
            mask = col_numeric <= value_numeric
        else:
            mask = pd.Series([True] * len(df))
    else:
        logger.warning(f"[Q&A Smart Retriever] Unknown filter op: {op}")
        return df

    return df[mask]


def _apply_filters_df(
    data: List[List[Any]],
    filters: List[Any],
) -> List[List[Any]]:
    """Apply multiple filters using pandas DataFrame (AND logic)."""
    df = _data_to_dataframe(data)
    if df is None or df.empty:
        return data

    original_count = len(df)

    for f in filters:
        try:
            field = (
                getattr(f, "field", None) if not isinstance(f, dict) else f.get("field")
            )
            op = getattr(f, "op", None) if not isinstance(f, dict) else f.get("op")
            val = (
                getattr(f, "value", None) if not isinstance(f, dict) else f.get("value")
            )

            if field and op:
                df = _apply_single_filter(df, field, op, val)
        except Exception as e:
            logger.warning(f"[Q&A Smart Retriever] Filter exception: {e}")

    logger.info(
        f"[Q&A Smart Retriever] Filter applied: {original_count} -> {len(df)} rows"
    )

    return _dataframe_to_data(df)


def _apply_aggregate_df(
    data: List[List[Any]],
    *,
    group_by: Optional[List[str]] = None,
    aggregations: Optional[Dict[str, str] | List[Aggregation]] = None,
    filters: Optional[List[Any]] = None,
    max_rows: int = 200,
    include_filtered_rows: bool = True,
) -> List[List[Any]]:
    """
    Apply aggregation using pandas DataFrame.

    Supports:
    - Pre-filtering before aggregation
    - Group by one or more columns
    - Multiple aggregation operations (sum, count, mean, min, max, first, last)
    - Alias for output column renaming

    aggregations can be:
    - Dict[str, str]: {"field": "sum", "field2": "count"}
    - List[Aggregation]: [Aggregation(field="x", op="sum", alias="total_x"), ...]
    """
    df = _data_to_dataframe(data)
    if df is None or df.empty:
        return data

    if not aggregations:
        return data

    original_count = len(df)
    original_header = data[0] if data else []

    # 1) Apply filters FIRST if provided
    if filters:
        for f in filters:
            try:
                field = (
                    getattr(f, "field", None)
                    if not isinstance(f, dict)
                    else f.get("field")
                )
                op = getattr(f, "op", None) if not isinstance(f, dict) else f.get("op")
                val = (
                    getattr(f, "value", None)
                    if not isinstance(f, dict)
                    else f.get("value")
                )
                if field and op:
                    df = _apply_single_filter(df, field, op, val)
            except Exception as e:
                logger.warning(
                    f"[Q&A Smart Retriever] Aggregate pre-filter exception: {e}"
                )

        logger.info(
            f"[Q&A Smart Retriever] Aggregate pre-filter: {original_count} -> {len(df)} rows"
        )

    if df.empty:
        return [original_header] if original_header else []

    filtered_count = len(df)

    # 2) Capture filtered rows sample BEFORE aggregation
    filtered_rows_sample: List[List[Any]] = []
    if include_filtered_rows and filtered_count > 0:
        sample_df = df.head(max_rows)
        filtered_rows_sample = _dataframe_to_data(sample_df)

    # 3) Normalize aggregations to dict format: {field: op}
    agg_dict: Dict[str, str] = {}
    alias_map: Dict[str, str] = {}

    if isinstance(aggregations, dict):
        # For example in Rheem Warranty --> Dict format: {"claims_usd_heat_pump": "sum"}
        agg_dict = {k: v for k, v in aggregations.items() if k in df.columns}
    elif isinstance(aggregations, list):
        for agg in aggregations:
            try:
                field = (
                    getattr(agg, "field", None)
                    if not isinstance(agg, dict)
                    else agg.get("field")
                )
                op = (
                    getattr(agg, "op", None)
                    if not isinstance(agg, dict)
                    else agg.get("op")
                )
                alias = (
                    getattr(agg, "alias", None)
                    if not isinstance(agg, dict)
                    else agg.get("alias")
                )

                if field and field in df.columns and op:
                    agg_dict[field] = op
                    if alias:
                        alias_map[field] = alias
            except Exception as e:
                logger.warning(f"[Q&A Smart Retriever] Aggregation parse error: {e}")

    if not agg_dict:
        logger.warning("[Q&A Smart Retriever] No valid aggregation columns found")
        return data

    # 4) Convert aggregation columns to numeric
    for col in agg_dict.keys():
        df[col] = pd.to_numeric(df[col], errors="coerce")

    try:
        # 5) Apply aggregation -- Should be not point 5 but not sure how to reorder these steps cleanly
        if group_by:
            valid_group_by = [g for g in group_by if g in df.columns]
            if not valid_group_by:
                logger.warning(
                    f"[Q&A Smart Retriever] No valid group_by columns: {group_by}"
                )
                valid_group_by = None
        else:
            valid_group_by = None

        if valid_group_by:
            result_df = df.groupby(valid_group_by, as_index=False).agg(agg_dict)
        else:
            agg_result = {}
            for col, op in agg_dict.items():
                if op == "sum":
                    agg_result[col] = [df[col].sum()]
                elif op == "count":
                    agg_result[col] = [df[col].count()]
                elif op == "mean":
                    agg_result[col] = [df[col].mean()]
                elif op == "min":
                    agg_result[col] = [df[col].min()]
                elif op == "max":
                    agg_result[col] = [df[col].max()]
                elif op == "first":
                    agg_result[col] = [df[col].iloc[0] if len(df) > 0 else None]
                elif op == "last":
                    agg_result[col] = [df[col].iloc[-1] if len(df) > 0 else None]
                else:
                    agg_result[col] = [None]
            result_df = pd.DataFrame(agg_result)

        # 6) Apply alias renaming OR auto-generate meaningful names
        # Build descriptive column names: field_operation (e.g., claims_usd_total_sum)
        rename_map = {}
        for col, op in agg_dict.items():
            if col in alias_map:
                rename_map[col] = alias_map[col]
            else:
                rename_map[col] = f"{col}_{op}"

        if rename_map:
            result_df = result_df.rename(columns=rename_map)

        agg_data = _dataframe_to_data(result_df)

        logger.info(
            f"[Q&A Smart Retriever] Aggregation: {original_count} rows -> filtered={filtered_count} -> agg_rows={len(result_df)} "
            f"(group_by={valid_group_by}, aggs={list(agg_dict.keys())})"
        )

        # 7) Combine filtered rows + aggregation result
        if include_filtered_rows and filtered_rows_sample:
            combined: List[List[Any]] = []
            combined.extend(filtered_rows_sample)
            agg_description = ", ".join(
                f"{op}({field})" for field, op in agg_dict.items()
            )
            group_description = (
                f"GROUP BY {', '.join(valid_group_by)}" if valid_group_by else "TOTAL"
            )
            separator_row = ["__AGGREGATE_RESULT__"] + [""] * (
                len(filtered_rows_sample[0]) - 1
            )
            combined.append(separator_row)
            # Metadata row with clear instructions
            meta_info = f"PRE-COMPUTED: {agg_description} {group_description} | filtered_rows={filtered_count} | DO NOT RECALCULATE"
            meta_row = [meta_info] + [""] * (len(filtered_rows_sample[0]) - 1)
            combined.append(meta_row)

            # Add aggregate data with descriptive headers
            combined.extend(agg_data)
            return combined
        else:
            return agg_data

    except Exception as e:
        logger.warning(f"[Q&A Smart Retriever] Aggregation failed: {e}")
        return data


class StructuredDataRetriever:
    """
    Engine-2: Given structured workflow outputs and a SemanticQueryPlan,
    return a list[PipelineStructuredOutput] containing only requested sections,
    and shaped by fields + record_plan.

    Contract:
    - mode=all: return section as-is (no field projection, no row filtering)
    - mode=none: return only header (or empty) to fail closed
    - mode=top_k: sort + slice
    - mode=filter: apply filters
    """

    def __init__(self, *, fail_open_on_bad_plan: bool = False):
        self.fail_open_on_bad_plan = fail_open_on_bad_plan

    def retrieve(
        self,
        *,
        structured_output_by_id: Dict[str, PipelineStructuredOutput],
        semantic_query_plan: SemanticQueryPlan,
    ) -> List[PipelineStructuredOutput]:
        plans = semantic_query_plan.sections
        if not plans:
            return []

        results: List[PipelineStructuredOutput] = []

        for sp in plans:
            sid = sp.id
            if not sid:
                continue

            so = structured_output_by_id.get(sid)
            if not so:
                logger.warning(
                    f"[Q&A Smart Retriever] Section not found in structured_output_by_id: {sid}"
                )
                continue

            fields = sp.fields or []
            record_plan = sp.record_plan
            rp_mode = (record_plan.mode or "none").strip().lower()

            # normalize mode
            if rp_mode not in RECORD_MODES:
                rp_mode = "none"

            # Work on a copy, never mutate original workflow output
            so_copy: PipelineStructuredOutput = copy.deepcopy(so)

            # Pull data
            data = getattr(so_copy, "data", None)
            if not isinstance(data, list) or not data or not isinstance(data[0], list):
                results.append(so_copy)
                continue

            # 1) Contract: mode=all => return section with max_rows cap and field projection
            if rp_mode == "all":
                max_rows = record_plan.max_rows or 200
                if len(data) > max_rows + 1:
                    data = data[: max_rows + 1]
                    logger.info(
                        f"[Q&A Smart Retriever] section={sid} mode=all capped to {max_rows} rows"
                    )

                if fields:
                    so_copy.data = _project_table_to_fields(data, fields)
                else:
                    so_copy.data = data

                results.append(so_copy)
                continue

            header = data[0]

            # 2) mode=none => fail closed: header only
            if rp_mode == "none":
                if record_plan.skip_reason:
                    logger.info(
                        f"[Q&A Smart Retriever] section={sid} mode=none "
                        f"(reason: {record_plan.skip_reason}) -- we don't use that LoL"
                    )
                so_copy.data = [header] if header else []
                results.append(so_copy)
                continue

            # 3) For filter/top_k: run record selection on ORIGINAL table (not projected)
            selected_rows_data = data

            if rp_mode == "filter":
                filters = record_plan.filters
                if not isinstance(filters, list) or not filters:
                    if self.fail_open_on_bad_plan:
                        selected_rows_data = data
                    else:
                        selected_rows_data = [header] if header else []
                else:
                    # Normal python filtering
                    # selected_rows_data = _apply_filters(data, filters)
                    # Use pandas-based filtering for robustness
                    selected_rows_data = _apply_filters_df(data, filters)

                    # Apply max_rows cap after filtering
                    # TODO: I am thinking should we need cap here for filter mode? Since we have aggregate mode so maybe we should cap here
                    max_rows = record_plan.max_rows or 200
                    if len(selected_rows_data) > max_rows + 1:
                        selected_rows_data = selected_rows_data[: max_rows + 1]
                        logger.info(
                            f"[Q&A Smart Retriever] section={sid} mode=filter capped to {max_rows} rows, actually No Rows == {len(selected_rows_data)}"
                        )

            elif rp_mode == "top_k":
                k = record_plan.k
                sort_by = record_plan.sort_by
                descending = bool(record_plan.descending)

                if (
                    not isinstance(k, int)
                    or not isinstance(sort_by, str)
                    or not sort_by
                ):
                    if self.fail_open_on_bad_plan:
                        selected_rows_data = data
                    else:
                        selected_rows_data = [header] if header else []
                else:
                    selected_rows_data = _apply_top_k(
                        data, k=k, sort_by=sort_by, descending=descending
                    )

            elif rp_mode == "aggregate":
                aggregations = record_plan.aggregations
                if not aggregations:
                    if self.fail_open_on_bad_plan:
                        selected_rows_data = data
                    else:
                        selected_rows_data = [header] if header else []
                else:
                    selected_rows_data = _apply_aggregate_df(
                        data,
                        group_by=record_plan.group_by,
                        aggregations=aggregations,
                        filters=record_plan.filters,
                        max_rows=record_plan.max_rows or 200,
                        include_filtered_rows=True,
                    )

            # 4) After record selection, project to fields (unless fields empty)
            if fields:
                final_data = _project_table_to_fields(selected_rows_data, fields)
            else:
                # If fields missing, fail closed unless fail_open
                if self.fail_open_on_bad_plan:
                    final_data = selected_rows_data
                else:
                    final_data = selected_rows_data[:1] if selected_rows_data else []

            so_copy.data = final_data
            results.append(so_copy)

        return results

class DeterministicFieldSelector:
    """
    Select fields using deterministic scoring.

    Scoring factors:
    1. Token overlap between question and field name
    2. Synonym matching
    3. Ontology metadata description matching
    4. Field importance heuristics (id, name, date fields)
    """

    def __init__(
        self,
        *,
        synonyms: Optional[Dict[str, Set[str]]] = None,
        token_weight: float = 0.3,
        synonym_weight: float = 0.2,
        description_weight: float = 0.3,
        importance_weight: float = 0.2,
        min_score_threshold: float = 0.1,
        max_fields: int = 20,
        always_include_patterns: Optional[List[str]] = None,
    ):
        self.synonyms = synonyms or DEFAULT_SYNONYMS
        self.token_weight = token_weight
        self.synonym_weight = synonym_weight
        self.description_weight = description_weight
        self.importance_weight = importance_weight
        self.min_score_threshold = min_score_threshold
        self.max_fields = max_fields

        self.always_include_patterns = always_include_patterns or [
            r"^id$",
            r"_id$",
            r"^name$",
            r"_name$",
            r"^key$",
            r"_key$",
            r"^code$",
            r"_code$",
            r"^sku$",
            r"^date$",
            r"_date$",
        ]
        self._always_include_re = [
            re.compile(p, re.IGNORECASE) for p in self.always_include_patterns
        ]

        self._reverse_synonyms: Dict[str, Set[str]] = {}
        for key, values in self.synonyms.items():
            if key not in self._reverse_synonyms:
                self._reverse_synonyms[key] = set()
            self._reverse_synonyms[key].update(values)
            for v in values:
                if v not in self._reverse_synonyms:
                    self._reverse_synonyms[v] = set()
                self._reverse_synonyms[v].add(key)
                self._reverse_synonyms[v].update(values - {v})

    @staticmethod
    def _tokenize(text: str) -> Set[str]:
        """Tokenize and normalize text."""
        if not text:
            return set()
        text = text.lower()
        tokens = re.findall(r"[a-z0-9]+", text)
        return set(t for t in tokens if len(t) > 1)

    def _get_synonyms(self, token: str) -> Set[str]:
        """Get synonyms for a token."""
        token = token.lower()
        synonyms = self._reverse_synonyms.get(token, set())
        return synonyms | {token}

    def _is_important_field(self, field_name: str) -> bool:
        """Check if field matches important patterns."""
        for pattern in self._always_include_re:
            if pattern.search(field_name):
                return True
        return False

    def _calculate_token_overlap(
        self,
        question_tokens: Set[str],
        field_name: str,
    ) -> float:
        """Calculate token overlap score."""
        field_tokens = self._tokenize(field_name)
        if not field_tokens or not question_tokens:
            return 0.0

        overlap = question_tokens & field_tokens
        if not overlap:
            return 0.0

        return len(overlap) / len(field_tokens)

    def _calculate_synonym_score(
        self,
        question_tokens: Set[str],
        field_name: str,
    ) -> float:
        """Calculate synonym matching score."""
        field_tokens = self._tokenize(field_name)
        if not field_tokens or not question_tokens:
            return 0.0

        expanded_question = set()
        for token in question_tokens:
            expanded_question.update(self._get_synonyms(token))

        overlap = expanded_question & field_tokens
        if not overlap:
            return 0.0

        return len(overlap) / len(field_tokens)

    def _calculate_description_score(
        self,
        question_tokens: Set[str],
        field_description: str,
    ) -> float:
        """Calculate description matching score."""
        if not field_description:
            return 0.0

        desc_tokens = self._tokenize(field_description)
        if not desc_tokens or not question_tokens:
            return 0.0

        expanded_question = set()
        for token in question_tokens:
            expanded_question.update(self._get_synonyms(token))

        overlap = expanded_question & desc_tokens
        if not overlap:
            return 0.0

        return min(1.0, len(overlap) / max(len(desc_tokens), 5))

    def _calculate_importance_score(self, field_name: str) -> float:
        """Calculate field importance score."""
        if self._is_important_field(field_name):
            return 1.0

        field_lower = field_name.lower()
        analytical_keywords = {
            "amount",
            "value",
            "total",
            "sum",
            "count",
            "quantity",
            "price",
            "cost",
            "revenue",
            "sales",
            "profit",
            "margin",
            "percent",
            "rate",
            "ratio",
            "average",
            "mean",
        }

        field_tokens = self._tokenize(field_lower)
        if field_tokens & analytical_keywords:
            return 0.5

        return 0.0

    def select_fields(
        self,
        question: str,
        allowed_fields: List[str],
        section_ontology: Optional[Dict[str, Any]] = None,
    ) -> Tuple[List[str], List[FieldScore]]:
        """
        Select relevant fields for a question.

        Returns:
            Tuple of (selected_fields, all_scores)
        """
        if not question or not allowed_fields:
            return [], []

        question_tokens = self._tokenize(question)
        if not question_tokens:
            return [], []

        metadata = {}
        if section_ontology and isinstance(section_ontology.get("metadata"), dict):
            metadata = section_ontology["metadata"]

        scores: List[FieldScore] = []

        for field_name in allowed_fields:
            field_info = metadata.get(field_name, {})
            if isinstance(field_info, dict):
                field_description = field_info.get("description", "")
            elif isinstance(field_info, str):
                field_description = field_info
            else:
                field_description = ""

            token_score = self._calculate_token_overlap(question_tokens, field_name)
            synonym_score = self._calculate_synonym_score(question_tokens, field_name)
            description_score = self._calculate_description_score(
                question_tokens, field_description
            )
            importance_score = self._calculate_importance_score(field_name)

            total = (
                self.token_weight * token_score
                + self.synonym_weight * synonym_score
                + self.description_weight * description_score
                + self.importance_weight * importance_score
            )

            scores.append(
                FieldScore(
                    field_name=field_name,
                    token_overlap_score=token_score,
                    synonym_score=synonym_score,
                    description_score=description_score,
                    importance_score=importance_score,
                    total_score=total,
                )
            )

        scores.sort(key=lambda s: s.total_score, reverse=True)

        selected = []
        selected_set = set()

        for field_name in allowed_fields:
            if self._is_important_field(field_name) and field_name not in selected_set:
                selected.append(field_name)
                selected_set.add(field_name)

        for score in scores:
            if len(selected) >= self.max_fields:
                break
            if score.field_name in selected_set:
                continue
            if score.total_score >= self.min_score_threshold:
                selected.append(score.field_name)
                selected_set.add(score.field_name)

        if not selected and scores:
            for score in scores[: min(5, len(scores))]:
                if score.field_name not in selected_set:
                    selected.append(score.field_name)
                    selected_set.add(score.field_name)

        logger.info(
            f"[DeterministicFieldSelector] Selected {len(selected)} fields "
            f"from {len(allowed_fields)} allowed"
        )

        return selected, scores

class DeterministicRecordPlanner:
    """
    Determine record plan using rule-based intent detection.

    Rules:
      - top/highest/lowest -- top_k
      - how many/total/sum/average -- aggregate
      - show me/list/which -- filter or top_k
      - small section (≤50 rows) or summarize_* -- all
      - specific entity mentioned -- filter (if valid_values available)
      - temporal reference - may need date filtering
    """

    def __init__(
        self,
        *,
        enable_aggregate_mode: bool = False,
        default_k: int = 20,
        small_section_threshold: int = 50,
        medium_section_threshold: int = 100,
    ):
        self.enable_aggregate_mode = enable_aggregate_mode
        self.default_k = default_k
        self.small_section_threshold = small_section_threshold
        self.medium_section_threshold = medium_section_threshold

        # Compile patterns
        self._ranking_patterns = [
            (re.compile(p, re.IGNORECASE), d) for p, d in RANKING_PATTERNS
        ]
        self._aggregate_patterns = [
            (re.compile(p, re.IGNORECASE), op) for p, op in AGGREGATE_PATTERNS
        ]
        self._listing_patterns = [
            re.compile(p, re.IGNORECASE) for p in LISTING_PATTERNS
        ]
        self._comparison_patterns = [
            re.compile(p, re.IGNORECASE) for p in COMPARISON_PATTERNS
        ]
        self._entity_patterns = [
            re.compile(p, re.IGNORECASE) for p in SPECIFIC_ENTITY_PATTERNS
        ]
        self._temporal_patterns = [
            (re.compile(p, re.IGNORECASE), t) for p, t in TEMPORAL_PATTERNS
        ]

    def detect_intent(self, question: str) -> IntentSignals:
        """Detect intent signals from question."""
        signals = IntentSignals()

        if not question:
            return signals

        question_lower = question.lower()

        # Detect ranking intent
        for pattern, direction in self._ranking_patterns:
            match = pattern.search(question_lower)
            if match:
                signals.is_ranking = True
                signals.ranking_direction = direction
                groups = match.groups()
                for g in groups:
                    if g and g.isdigit():
                        signals.suggested_k = min(int(g), 100)
                        break
                break

        # Extract K from other patterns
        k_match = K_EXTRACTION_PATTERN.search(question_lower)
        if k_match:
            for g in k_match.groups():
                if g and g.isdigit():
                    signals.suggested_k = min(int(g), 100)
                    break

        # Detect aggregate intent
        for pattern, op in self._aggregate_patterns:
            if pattern.search(question_lower):
                signals.is_aggregate = True
                if op not in signals.aggregate_ops:
                    signals.aggregate_ops.append(op)

        # Detect listing intent
        for pattern in self._listing_patterns:
            if pattern.search(question_lower):
                signals.is_listing = True
                break

        # Detect comparison intent
        for pattern in self._comparison_patterns:
            if pattern.search(question_lower):
                signals.is_comparison = True
                break

        # Detect specific entities (use original case for entity extraction)
        for pattern in self._entity_patterns:
            matches = pattern.findall(question)
            for match in matches:
                entity = match.strip() if isinstance(match, str) else match[0].strip()
                if entity and len(entity) > 1:
                    # Clean up the entity
                    entity = re.sub(r"\s+", " ", entity).strip()
                    if entity and entity not in signals.specific_entities:
                        signals.specific_entities.append(entity)
                        signals.is_specific = True

        # Detect temporal references
        for pattern, temporal_type in self._temporal_patterns:
            matches = pattern.findall(question_lower)
            for match in matches:
                if isinstance(match, tuple):
                    # Join tuple parts (e.g., ("Jan", "2024") -> "Jan 2024")
                    temporal_ref = " ".join(str(m) for m in match if m).strip()
                else:
                    temporal_ref = str(match).strip()

                if temporal_ref and temporal_ref not in signals.temporal_references:
                    signals.temporal_references.append(temporal_ref)
                    signals.is_temporal = True

        logger.debug(
            f"[DeterministicRecordPlanner] Detected intent: "
            f"ranking={signals.is_ranking}, aggregate={signals.is_aggregate}, "
            f"listing={signals.is_listing}, comparison={signals.is_comparison}, "
            f"specific={signals.is_specific}, temporal={signals.is_temporal}, "
            f"entities={signals.specific_entities}, temporal_refs={signals.temporal_references}"
        )

        return signals

    def _infer_sort_field(
        self,
        selected_fields: List[str],
        intent: IntentSignals,
    ) -> Optional[str]:
        """Infer best sort field from selected fields."""
        if not selected_fields:
            return None

        sort_priority_keywords = [
            "amount",
            "value",
            "total",
            "sum",
            "revenue",
            "sales",
            "profit",
            "cost",
            "price",
            "count",
            "quantity",
            "percent",
            "rate",
            "ratio",
            "margin",
            "score",
            "impact",
            "variance",
        ]

        for keyword in sort_priority_keywords:
            for fld in selected_fields:
                if keyword in fld.lower():
                    return fld

        numeric_patterns = [
            "amount",
            "value",
            "total",
            "count",
            "qty",
            "num",
            "pct",
            "percent",
        ]
        for fld in selected_fields:
            fld_lower = fld.lower()
            if any(p in fld_lower for p in numeric_patterns):
                return fld

        return selected_fields[0] if selected_fields else None

    def _infer_aggregate_field(
        self,
        selected_fields: List[str],
        intent: IntentSignals,
    ) -> Optional[str]:
        """Infer best field for aggregation."""
        if not selected_fields:
            return None

        if "count" in intent.aggregate_ops:
            return selected_fields[0]

        numeric_keywords = [
            "amount",
            "value",
            "total",
            "revenue",
            "sales",
            "profit",
            "cost",
            "price",
            "quantity",
            "count",
            "margin",
        ]

        for keyword in numeric_keywords:
            for fld in selected_fields:
                if keyword in fld.lower():
                    return fld

        return selected_fields[0]

    def _infer_filter_field(
        self,
        selected_fields: List[str],
        allowed_fields: List[str],
        intent: IntentSignals,
    ) -> Optional[Tuple[str, str]]:
        if not intent.specific_entities:
            return None
        filter_field_keywords = [
            "name",
            "region",
            "store",
            "location",
            "state",
            "city",
            "category",
            "type",
            "class",
            "group",
            "segment",
            "dc",
            "warehouse",
            "sku",
            "product",
            "model",
        ]
        entity = intent.specific_entities[0]
        for keyword in filter_field_keywords:
            for fld in allowed_fields:
                if keyword in fld.lower():
                    return (fld, entity)

        return None

    def _infer_date_field(
        self,
        allowed_fields: List[str],
    ) -> Optional[str]:
        date_keywords = ["date", "time", "period", "month", "year", "quarter"]
        for keyword in date_keywords:
            for fld in allowed_fields:
                if keyword in fld.lower():
                    return fld
        return None

    def plan_records(
        self,
        question: str,
        section_id: str,
        selected_fields: List[str],
        allowed_fields: List[str],
        row_count: int,
    ) -> Dict[str, Any]:
        """
        Generate record plan based on rules.
        """
        intent = self.detect_intent(question)

        logger.debug(
            f"[DeterministicRecordPlanner] section={section_id}, "
            f"row_count={row_count}, intent={intent}"
        )

        # Rule 1: Small sections or summary tables -- all
        if row_count <= self.small_section_threshold:
            return {"mode": "all", "max_rows": row_count}

        if section_id.startswith("summarize_"):
            return {"mode": "all", "max_rows": min(row_count, 200)}

        # Rule 2: Ranking questions → top_k
        if intent.is_ranking:
            sort_field = self._infer_sort_field(selected_fields, intent)
            return {
                "mode": "top_k",
                "k": intent.suggested_k or self.default_k,
                "sort_by": sort_field,
                "descending": intent.ranking_direction == "desc",
            }

        # Rule 3: Aggregate questions -- aggregate (if enabled)
        if intent.is_aggregate and self.enable_aggregate_mode:
            agg_field = self._infer_aggregate_field(selected_fields, intent)
            agg_op = intent.aggregate_ops[0] if intent.aggregate_ops else "count"

            plan = {
                "mode": "aggregate",
                "aggregations": [
                    {
                        "field": agg_field,
                        "op": agg_op,
                        "alias": f"{agg_field}_{agg_op}" if agg_field else agg_op,
                    }
                ],
            }

            # Add group_by if specific entity detected (for breakdown)
            if intent.is_specific:
                filter_info = self._infer_filter_field(
                    selected_fields, allowed_fields, intent
                )
                if filter_info:
                    plan["group_by"] = [filter_info[0]]

            return plan

        # Rule 3.5: Specific entity + listing = filter mode
        if intent.is_specific and intent.is_listing:
            filter_info = self._infer_filter_field(
                selected_fields, allowed_fields, intent
            )
            if filter_info:
                field_name, filter_value = filter_info
                return {
                    "mode": "filter",
                    "filters": [
                        {
                            "field": field_name,
                            "op": "contains",
                            "value": filter_value,
                        }
                    ],
                }

        # Rule 4: Listing questions===> top_k (conservative)
        if intent.is_listing:
            sort_field = self._infer_sort_field(selected_fields, intent)
            return {
                "mode": "top_k",
                "k": intent.suggested_k or self.default_k,
                "sort_by": sort_field,
                "descending": True,
            }

        # Rule 5: Medium sections → all with cap
        if row_count <= self.medium_section_threshold:
            return {"mode": "all", "max_rows": row_count}

        # Rule 6: Default --- top_k
        sort_field = self._infer_sort_field(selected_fields, intent)
        return {
            "mode": "top_k",
            "k": self.default_k,
            "sort_by": sort_field,
            "descending": True,
        }

    def get_confidence(self, intent: IntentSignals) -> float:
        base_confidence = 0.5
        if intent.is_ranking:
            base_confidence = 0.9
        elif intent.is_aggregate:
            base_confidence = 0.85
        elif intent.is_listing:
            base_confidence = 0.7
        if intent.is_specific and intent.specific_entities:
            base_confidence = min(base_confidence + 0.05, 0.95)
        if intent.is_temporal:
            base_confidence = min(base_confidence + 0.03, 0.95)
        return base_confidence

class DeterministicSearchIndex:

    def __init__(

    def _get_embedding_model(self):

    def _build_document_text(
        self,
        sid: str,
        ontology_item: Dict[str, Any],
    ) -> SearchDocument:
        """Build searchable document from ontology item."""
        title = ontology_item.get("title", "") or ""
        description = ontology_item.get("nlp_methodology_description", "") or ""

        metadata = ontology_item.get("metadata", {})
        metadata_parts = []
        field_names = []

        if isinstance(metadata, dict):
            for field_name, field_info in metadata.items():
                field_names.append(field_name)
                metadata_parts.append(field_name)
                if isinstance(field_info, dict):
                    desc = field_info.get("description", "")
                    if desc:
                        metadata_parts.append(str(desc))
                    # Include valid_values for better matching
                    valid_values = field_info.get("valid_values", [])
                    if valid_values and isinstance(valid_values, list):
                        metadata_parts.extend([str(v) for v in valid_values[:20]])
                elif isinstance(field_info, str):
                    metadata_parts.append(field_info)

        # Extract graph relationships if available
        graph = ontology_item.get("graph", {})
        graph_parts = []
        if isinstance(graph, dict):
            entities = graph.get("entities", [])
            if isinstance(entities, list):
                for ent in entities[:20]:
                    if isinstance(ent, dict):
                        graph_parts.append(ent.get("name", ""))
                    elif isinstance(ent, str):
                        graph_parts.append(ent)

        metadata_text = " ".join(metadata_parts)
        graph_text = " ".join(graph_parts)
        all_text = f"{sid} {title} {description} {metadata_text} {graph_text}"

        return SearchDocument(
            sid=sid,
            title=title,
            description=description,
            metadata_text=metadata_text,
            all_text=all_text,
            field_names=field_names,
        )

    def build_index(self, ontology: Dict[str, Any]) -> None:
        """Build search index from ontology."""
        if not ontology:
            return

        self.documents.clear()
        self._sid_list.clear()
        self._tokenized_corpus.clear()
        self._embeddings.clear()
        self._bm25_index = None

        for sid, item in ontology.items():
            if not isinstance(item, dict):
                continue

            doc = self._build_document_text(sid, item)
            self.documents[sid] = doc
            self._sid_list.append(sid)

            # Tokenize for BM25
            tokens = self._tokenize(doc.all_text)
            self._tokenized_corpus.append(tokens)

        # Build BM25 index using rank_bm25 (same as similarity.py)
        if self._tokenized_corpus:
            self._bm25_index = BM25Okapi(self._tokenized_corpus)
            logger.info(
                f"[DeterministicSearchIndex] Built BM25 index with {len(self._tokenized_corpus)} documents"
            )

        if self.enable_embeddings:
            self._build_embedding_index()

        logger.info(
            f"[DeterministicSearchIndex] Built index: {len(self.documents)} docs, "
            f"embeddings={self.enable_embeddings}"
        )

    def _build_embedding_index(self) -> None:
        model = self._get_embedding_model()
        if model is None:
            return

        try:
            texts = []
            sids = []
            for sid, doc in self.documents.items():
                texts.append(doc.all_text[:1000])  # Truncate for efficiency
                sids.append(sid)

            if texts:
                embeddings = model.encode(texts, show_progress_bar=False)
                for sid, emb in zip(sids, embeddings):
                    self._embeddings[sid] = emb.tolist()

                logger.info(
                    f"[DeterministicSearchIndex] Built embeddings for {len(self._embeddings)} documents"
                )
        except Exception as e:
            logger.warning(
                f"[DeterministicSearchIndex] Failed to build embedding index: {e}"
            )
            self.enable_embeddings = False

    def _bm25_scores(self, query: str) -> Dict[str, float]:
        """
        Calculate BM25 scores for all documents using rank_bm25.
        """
        if self._bm25_index is None or not self._sid_list:
            return {}

        query_tokens = self._tokenize(query)
        if not query_tokens:
            return {}

        scores = self._bm25_index.get_scores(query_tokens)
        # Map scores back to sids
        return {sid: float(score) for sid, score in zip(self._sid_list, scores)}

    def _embedding_score(self, query: str, sid: str) -> float:
        """Calculate embedding similarity score."""
        if not self.enable_embeddings:
            return 0.0

        model = self._get_embedding_model()
        if model is None:
            return 0.0

        if sid not in self._embeddings:
            return 0.0

        try:
            query_emb = model.encode([query[:1000]], show_progress_bar=False)[0]
            doc_emb = self._embeddings[sid]

            # Cosine similarity
            dot_product = sum(a * b for a, b in zip(query_emb, doc_emb))
            norm_q = math.sqrt(sum(a * a for a in query_emb))
            norm_d = math.sqrt(sum(a * a for a in doc_emb))

            if norm_q > 0 and norm_d > 0:
                return dot_product / (norm_q * norm_d)
            return 0.0
        except Exception as e:
            logger.warning(f"[DeterministicSearchIndex] Embedding score error: {e}")
            return 0.0

    def search(
        self,
        query: str,
        top_k: int = 10,
        min_score: float = 0.0,
    ) -> List[SearchResult]:
        """
        Search for relevant sections.

        Returns union of top_k lexical + top_k embedding results, reranked.
        """
        if not query or not self.documents:
            return []

        results: Dict[str, SearchResult] = {}

        # Get BM25 scores for all documents
        bm25_scores = self._bm25_scores(query)

        for sid, lexical_score in bm25_scores.items():
            results[sid] = SearchResult(sid=sid, lexical_score=lexical_score)

        # Get top-k by lexical score
        lexical_top = sorted(
            results.values(), key=lambda r: r.lexical_score, reverse=True
        )[:top_k]

        # Add embedding scores if enabled
        if self.enable_embeddings:
            embedding_candidates = set(r.sid for r in lexical_top)

            # Also consider some documents not in lexical top
            if len(self._embeddings) > len(embedding_candidates):
                remaining = set(self._embeddings.keys()) - embedding_candidates
                embedding_candidates.update(list(remaining)[:top_k])

            for sid in embedding_candidates:
                if sid not in results:
                    results[sid] = SearchResult(sid=sid)
                results[sid].embedding_score = self._embedding_score(query, sid)

        # Calculate combined scores
        for result in results.values():
            if self.enable_embeddings:
                result.combined_score = (
                    self.lexical_weight * result.lexical_score
                    + self.embedding_weight * result.embedding_score
                )
            else:
                result.combined_score = result.lexical_score

        # Sort by combined score and filter
        final_results = sorted(
            results.values(), key=lambda r: r.combined_score, reverse=True
        )
        final_results = [r for r in final_results if r.combined_score >= min_score]

        return final_results[:top_k]

    def get_document(self, sid: str) -> Optional[SearchDocument]:
        return self.documents.get(sid)
